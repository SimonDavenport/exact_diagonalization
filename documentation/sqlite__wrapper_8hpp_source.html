72  //!
73  //! Template method setting explicit data values
74  //!
75  template<typename T>
76  void SetValues(const std::string name, const T data)
77  {
78  m_name = name;
79  // Convert template typename to a string containing
80  // the SQL type name and convert the input data into a string
81  std::stringstream tmp;
82  tmp.str("");
83  tmp<<data;
84  m_data = tmp.str();
85  if(is_same<T, int>::value)
86  {
87  // Set INT type if a type is not already defined by the constructor
88  if("" == m_type) m_type = "INT";
89  assert(m_type =="INT");
90  }
91  else if(is_same<T, double>::value)
92  {
93  // Set REAL type if a type is not already defined by the constructor
94  if("" == m_type) m_type = "REAL";
95  assert(m_type =="REAL");
96  }
97  else
98  {
99  // Set TEXT type if a type is not already defined by the constructor
100  if("" == m_type) m_type = "TEXT";
101  // Allow BLOB type also here
102  assert(m_type =="TEXT" || m_type == "BLOB");
103  }
104  }
