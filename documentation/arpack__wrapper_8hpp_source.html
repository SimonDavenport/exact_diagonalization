234  ////////////////////////////////////////////////////////////////////////////////
235  //! \brief This function calls an ARPACK diagonalization routine to obtain a
236  //! approximate eigenvalues/eigenvectors of a complex hermitian matrix
237  //! or real symmetric matrix.
238  //!
239  //! Returns selected Ritz values/vectors (approximate eigenvalues/eigenvectors)
240  //! to the memory address specified by the eigenvalues/eigenvectors variables.
241  //!
242  //! NOTE: on shift and invert mode
243  //!
244  //! Instead of solving Ax = lambda x we solve (A-shift*I)^-1 = nu x
245  //! Then use that nu = 1/(lambda - shift)
246  //!
247  //! This approach reduces the number of ARPACK iterations required to
248  //! converge to low lying eigenvalues of the original spectrum, at the
249  //! expense of a more computationally demanding linear solve step at
250  //! each iteration and an initial Cholesky decomposition of the
251  //! matrix (which can take a lot of memory if the decomposition is at
252  //! lot less sparse than the original matrix.) The shift value must be
253  //! chosen close to the lowest lying eigenvalue for the method to work.
254  ////////////////////////////////////////////////////////////////////////////////
255  template <typename T>
256  bool DiagonalizeSymmetricMatrix(
257  const std::function<void(T* inVector, T* outVector)>& MatrixVectorMultiply,
258  //!< Function to perform matrix-vector operation
259  T* eigenvectors, //!< A memory address (N by M in size) to store
260  //! the resulting eigenvectors
261  double* eigenvalues, //!< The memory address (N in size) to store
262  //! the resulting eigenvalues
263  const int nodeDim, //!< Dimension of the matrix/eigenvectors
264  //! stored on each node
265  const int fullDim, //!< Full dimension of the matrix
266  const int nbrEigenvalues) //!< Number of eigenvalues requested
267  {
268  ////// A DESCRIPTION OF ALL ?naupd_ ARGUMENTS ///////////////////
269 
270  int IDO; //!< Must be set to zero initially,
271  //! afterwards specifies the type of operation performed
272  //! and value 99 tells us the iterative procedure has finished
273  char BMAT; //!< 'I' sets a standard eigenvalue problem, 'G' sets a
274  //! generalized eigenvalue problem instead
275  int N; //!< Dimension of the eigenvalue problem
276  char WHICH[2]; //!< Option for range of eigenvalue returned:
277  //! 'LM' -> eigenvalues of largest magnitude
278  //! 'SM' -> eigenvalues of smallest magnitude
279  //! 'LR' -> eigenvalues of largest real part
280  //! 'SR' -> eigenvalues of smallest real part
281  //! 'LI' -> eigenvalues of largest imaginary part
282  //! 'SI' -> eigenvalues of smallest imaginary part
283  int NEV; //!< Number of eigenvalues/eigenvectors to find
284  double TOL; //!< Stopping tolerance (defaults to machine precision)
285  T* RESID; //!< A dimension-N vector containing the residuals
286  int NCV; //!< Specify the number of Arnoldi vectors that are
287  //! generated with each iteration (must lie between
288  //! 2+NEV and N with 2*NEV+1 recommended)
289  T* V; //!< Memory space to store the Arnoldi vectors (array
290  //! size N by NCV)
291  int LDV; //!< Leading dimension of the Arnodli vectors
292  int IPARAM[11]; //!< Sets 11 algorithm parameters:
293  //! 0: Set implicit shifts
294  //! - 0 user provided
295  //! - 1 exact shifts from the current Hessenberg matrix
296  //! - 2 not defined
297  //! 1: Not referenced
298  //! 2: Maximum number of iterations allowed (returns the
299  //! actual number)
300  //! 3: Block size used in recurrence (only works for 1)
301  //! 4: Specify the number of eigenvalues to include in
302  //! the convergence criteria
303  //! 5: Not referenced
304  //! 6: Sets the problem class
305  //! - 1: Ax = lambda x
306  //! - 2: Ax = lambda M x (M positive definite)
307  //! - 3: Ax = lambda M x (M positive semi-definite)
308  //! 7: Holds the number of shifts that the user provides
309  //! (only used if IPARAM[0] = 2)
310  //! 8: Counts the total number of matrix-vector multiplications
311  //! 9: Counts the total matrix-vector multiplications in the
312  //! generalized eigenvalue problem (BMAT = 'G')
313  //! 10: Counts the number of re-orthogonalization steps
314  int IPNTR[14]; //!< Pointer to mark starting locations for different
315  //! output types stored in WORKD and WORKL arrays:
316  //! 0: Pointer to current right-vector X in WORKD
317  //! 1: Pointer to current left-vector Y in WORKD
318  //! 2: Pointer to B*X when used in IPARAM 6: 3 mode
319  //! 3: Pointer to next available WORKL location
320  //! 4: Pointer to upper Hessenberg matrix
321  //! 5: Pointer to Ritz array
322  //! 6: Pointer to projected Ritz array
323  //! 7: Pointer to error bounds in WORKL
324  //! 8: Pointer to original Ritz values
325  //! 9: Not used
326  //! 10: Pointer to error bounds
327  //! 11: Pointer to shifts in WORKL
328  T* WORKD; //!< Working array of dimension 3*N
329  T* WORKL; //!< Working array of dimension LWORKL
330  int LWORKL; //!< Dimension of WORKL must be at least 3*NCV**2 + 5*NCV
331  double* RWORK; //!< Working array of dimension NCV
332  int INFO; //! Contains output error codes
333  ////// A DESCRIPTION OF ADDITIONAL ?neupd_ ARGUMENTS //////////
334  int RECV; //!< FALSE: compute Ritz eigenvalues only; TRUE: also
335  //! compute Ritz vectors
336  char HOWMNY; //!< 'A': compute all NEV Ritz vectors
337  //! 'P': compute all NEV Schur vectors
338  int* SELECT; //!< Internal work space
339  T* D; //!< Contains the output Ritz eigenvalues
340  T* DR; //!< Contains the output Ritz eigenvalues
341  T* DI; //!< Contains the output Ritz eigenvalues
342  T* Z; //!< If RECV = true, contains Ritz eigenvectors.
343  //! Must be of dimension N*NEV
344  int LDZ; //!< Leading dimension of the Z array
345  T* WORKEV; //!< Working space of dimension 2*NCV
346  T SIGMA; //!< If IPARAM 6: is 3 then contains the "shift value"
347  T SIGMAR; //!< If IPARAM 6: is 3 then contains the "shift value"
348  T SIGMAI; //!< If IPARAM 6: is 3 then contains the "shift value"
349  // Remaining arguments are identical to the (p)znaupd_ list
350  ////// ADDITIONAL HIDDEN CHAR LENGTH ARGUMENTS ////////////////////
351  int LEN_BMAT = 1; //!< BMAT is a length 1 character
352  int LEN_WHICH = 2; //!< WHICH is a length 2 character
353  int LEN_HOWMNY = 1; //!< HOWMNY is a length 1 character
354  ////// ITERATION COUNTER VARIABLES //////////
355  int iterationCounter = 0; //!< Count the number of ARPACK iterations taken
356  IDO = 0; // Set value for initial iteration
357  BMAT = m_typeOfProblem;
358  N = nodeDim;
359  WHICH[0] = m_eigenvalueMagnitude;
360  WHICH[1] = m_whatMagnitude;
361  NEV = nbrEigenvalues;
362  NCV = std::min(fullDim, 2*NEV+1+20);
363  LDV = N;
364  IPARAM[0] = 1; // Always set ARPACK provided shifts
365  IPARAM[2] = m_maxIterations;
366  IPARAM[3] = m_blockSize;
367  IPARAM[4] = nbrEigenvalues;
368  IPARAM[6] = 1; // Always use ARPACK in its standard mode
369  if(utilities::is_same<T, dcmplx>::value)
370  {
371  LWORKL = (3*NCV+5)*NCV;
372  }
373  else if(utilities::is_same<T, double>::value)
374  {
375  LWORKL = (3*NCV+6)*NCV;
376  }
377  INFO = m_provideInitial;
378  RECV = 1; // Compute eigenvectors
379  HOWMNY = 'A'; // Alternative options not implemented
380  LDZ = N;
381  SIGMA = 0; // Set the initial shift value, ARPACK will do the rest
382  SIGMAR = 0;
383  SIGMAI = 0;
384  ////// INITIALIZE MEMORY ALLOCATION ////////////////////////////
385  // Print a warning if we're about to allocate a LOT of working memory
386  uint64_t memoryRequirement = (uint64_t)((NEV+1+(1+NCV+3)*N+LWORKL)*sizeof(T));
387  if(memoryRequirement>memoryQueryLimit)
388  {
389  std::cerr<<"\n\tDiagonalizeSymmetricMatrix WARNING: about to allocate "
390  <<memoryRequirement/(1024.0*1024.0)<<" MB for znaupd_ routine."
391  <<"\n\tPRESS ANY KEY TO CONTINUE."<<std::endl;
392  getchar();
393  }
394  ////// ALLOCATE MEMORY ////////////////////////////////////////////
395  RESID = new T[N];
396  V = new T[N*NCV];
397  WORKD = new T[3*N];
398  WORKL = new T[LWORKL];
399  SELECT = new int[NCV];
400  if(utilities::is_same<T, dcmplx>::value)
401  {
402  RWORK = new double[NCV];
403  D = new T[NEV+1];
404  }
405  else if(utilities::is_same<T, double>::value)
406  {
407  DR = new T[NEV+1];
408  DI = new T[NEV+1];
409  }
410  Z = eigenvectors; // Set to external memory address
411  WORKEV = new T[2*NCV];
412  ////// SET RESID TO THE STARTING VECTOR, IF ONE IS PROVIDED ////////
413  if(m_provideInitial)
414  {
415  std::ifstream f_initVector;
416  f_initVector.open(m_initialVectorFile.c_str(), std::ios::binary);
417  if(!f_initVector.is_open())
418  {
419  std::cerr<<"\n\tERROR in DiagonalizeSymmetricMatrix: Cannot open initial vector file "<<m_initialVectorFile<<std::endl;
420  std::cerr<<"\n\tUsing random starting vector instead."<<std::endl;
421  INFO = 0;
422  }
423  else
424  {
425  f_initVector.read(reinterpret_cast<char*>(RESID), (long int)N*sizeof(T));
426  f_initVector.close();
427  }
428  }
429  ////// RUN ITERATIVE LANCZOS ALGORITHM //////////////////////////
430  // Recursively call the (p)znaupd_ routine until the desired eigenvalues
431  // have properly converged
432  utilities::cout.AdditionalInfo()<<std::endl;
433  do
434  {
435  utilities::cout.AdditionalInfo()<<"\t ARPACK ITERATION "<<iterationCounter<<"\r";
436  fflush(stdout);
437  if(utilities::is_same<T, dcmplx>::value)
438  {
439  znaupd_(&IDO, &BMAT, &N, &WHICH[0], &NEV, &TOL, RESID, &NCV, V,
440  &LDV, IPARAM, IPNTR, WORKD, WORKL, &LWORKL, RWORK, &INFO,
441  &LEN_BMAT, &LEN_WHICH);
442  }
443  else if(utilities::is_same<T, double>::value)
444  {
445  dnaupd_(&IDO, &BMAT, &N, &WHICH[0], &NEV, &TOL, RESID, &NCV, V,
446  &LDV, IPARAM, IPNTR, WORKD, WORKL, &LWORKL, &INFO,
447  &LEN_BMAT, &LEN_WHICH);
448  }
449  if(INFO<0)
450  {
451  if(utilities::is_same<T, dcmplx>::value)
452  {
453  std::cerr<<"\n\tERROR WITH znaupd_"
454  <<"\n\t ERROR CODE WAS "<<INFO<<" (see http://www.mathkeisan.com/usersguide/man/znaupd.html)"
