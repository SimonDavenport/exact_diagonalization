111  //!
112  //! Generate a Fock basis
113  //!
114  void SphereTwoLevelPseudopotentialModel::BuildFockBasis(
115  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
116  {
117  if(0 == mpi.m_id) // FOR THE MASTER NODE
118  {
119  if(m_params.m_blockDiagonalize)
120  {
121  utilities::cout.MainOutput()<<"\n\t============ BUILDING FOCK BASIS (lz_tot = "<<m_params.m_totalLz<<") ============ "<<std::endl;
122  }
123  else
124  {
125  utilities::cout.MainOutput()<<"\n\t============ BUILDING FULL FOCK BASIS ============ "<<std::endl;
126  }
127  }
128  // The function "GenerateFockSpace" requires a single argument:
129  // a function that takes a Fock basis state as an argument and returns
130  // a bool saying whether that state is in the specified sector or not.
131  bool isZeroDimensional = false;
132  iSize_t nbrOrbitalsLLL = (m_params.m_nbrOrbitals-2)/2;
133  if(m_params.m_blockDiagonalize)
134  {
135  TwoLevelFermionFockBasis basis;
136  // To make the function we're going to use std::bind to fix parameters
137  // in the function called TestLinearMomentumSector defined above, such
138  // that only the first parameter (the Fock state) can be
139  // specified and all other parameters are fixed.
140  isZeroDimensional = basis.GenerateFockSpace(m_params.m_nbrParticles, nbrOrbitalsLLL, nbrOrbitalsLLL+2,
141  std::bind(&TestAngularMomentumSector2LL, std::placeholders::_1,
142  m_params.m_totalLz, m_params.m_maxLz2), mpi);
143  m_hamiltonian.SetFockBasis(basis);
144  }
145  else
146  {
147  TwoLevelFermionFockBasis basis;
148  // If we're not block diagonalizing then we pass a dummy function
149  // to m_hamiltonian (the function is declared as a "lambda function")
150  // This function will always return true.
151  isZeroDimensional = basis.GenerateFockSpace(m_params.m_nbrParticles, nbrOrbitalsLLL, nbrOrbitalsLLL+2,
152  [](const TwoLevelFockState& state){return true;}, mpi);
153  m_hamiltonian.SetFockBasis(basis);
154  }
155 
156  if(isZeroDimensional)
157  {
158  if(0 == mpi.m_id) // FOR THE MASTER NODE
159  {
160  utilities::cout.SecondaryOutput()<<"\n\t============ FOCK BASIS IS ZERO DIMENSIONAL ============ "<<std::endl;
161  }
162  m_params.m_fockBasisBuilt = false;
163  return;
164  }
165  m_params.m_fockBasisBuilt = true;
166  }

182  ////////////////////////////////////////////////////////////////////////////////
183  //! \brief Generate a matrix representation of our Hamiltonian in the 2D
184  //! momentum space basis.
185  //!
186  //! When called multiple times, this function will not construct a new
187  //! unless the previous Hamiltonian has been cleared with the
188  //! ClearHamiltonian function.
189  ////////////////////////////////////////////////////////////////////////////////
190  void SphereTwoLevelPseudopotentialModel::BuildHamiltonian(
191  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper
192  {
193  if(m_params.m_termTablesBuilt && m_params.m_fockBasisBuilt && !m_params.m_hamiltonianBuilt)
194  {
195  ////// GENERATE THE HAMILTONIAN //////
196  if(0 == mpi.m_id) // FOR THE MASTER NODE
197  {
198  utilities::cout.MainOutput()<<"\n\t============ BUILDING HAMILTONIAN ============ "<<std::endl;
199  }
200  m_hamiltonian.Initialize(m_params.m_nbrParticles, m_params.m_nbrOrbitals, utilities::_SPARSE_MAPPED_, mpi);
201  MPI_Barrier(mpi.m_comm);
202  ////// Build the Hamiltonian
203  if(m_quadraticTables.GetDimension()>0)
204  {
205  m_hamiltonian.Add_CdC_Terms(&m_quadraticTables, 0, 0, 0, m_hamiltonian.m_data.m_fockSpaceDim, mpi);
206  }
207  if(m_quadraticTables2LL.GetDimension()>0)
208  {
209  m_hamiltonian.Add_CdC_Terms(&m_quadraticTables2LL, 1, 1, 0, m_hamiltonian.m_data.m_fockSpaceDim, mpi);
210  }
211  m_hamiltonian.Add_CdCdCC_Terms(&m_quarticTables, 0, 0, 0, 0, 0, m_hamiltonian.m_data.m_fockSpaceDim, mpi);
212  m_hamiltonian.Add_CdCdCC_Terms(&m_quarticTables2LL, 1, 1, 1, 1, 0, m_hamiltonian.m_data.m_fockSpaceDim, mpi);
213  m_hamiltonian.PrintMemoryAllocation(mpi);
214  m_hamiltonian.PrintHamiltonian(0, mpi);
215  m_params.m_hamiltonianBuilt = true;
216  }
217  else
218  {
219  m_params.m_hamiltonianBuilt = false;
220  }
221  }
