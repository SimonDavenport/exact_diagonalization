5 //! \file
6 //! This file contains a class to implement sparse array storage
7 //! using a hash table scheme.
8 //!
9 //! NOTES:
10 //!
11 //! The SPARSE MAPPED type storage scheme is a wrapper using the following
12 //! classes. The class used can be set by a precompiler definition:
13 //!
14 //! 1. std::unordered_map implements hash table data structure (c++ 11
15 //! is required). It is relatively slow to populate compared to
16 //! other implementations and there is a high memory overhead.
17 //! To add insult to injury, the memory clear function does not
18 //! reliably release memory resources back to the operating system. Avoid!
19 //!
20 //! 2. The sparsehash/sparse_hash_map library is highly memory efficient
21 //! with a 2-bit per element overhead.
22 //!
23 //! 3. The sparsehash/dense_hash_map library is highly speed efficient
24 //! but with a much higher memory overhead compared to the sparse
25 //! implementation (it's still better than std::map however).
26 //!
27 //! Both libraries 2 and 3 reliably release memory resources when the resize(0)
28 //! function is called. One disadvantage is that you need to reserve a map key
29 //! to store deleted entries (also an empty map key in the dense map case)
30 //! For convenience I choose the last possible map key
31 //! (e.g. the highest int or long int value) to be the deleted
32 //! entry key and (last possible value -1) for the empty map key.
33 //! Another issue is that the map is unordered, and it must be sorted if
34 //! converted to another format e.g. CRS.
35 //!
36 //! http://research.neustar.biz/2011/11/27/big-memory-part-3-5-google-sparsehash/

81  //! \brief Comparison operator to test if two CrsSparseMatrix data structures
82  //! are identical up to a given tolerence. Only classes with the same template
83  //! arguments can be compared.
84  //!
85  //! Note that the comparison algorithm requires an in-place sort of the
86  //! underlying data and column arrays in the CrsSparseMatrix class. This
87  //! does not change the data represented.
88  //////////////////////////////////////////////////////////////////////////////////
89  template<typename T>
90  bool SameTest(
91  MappedSparseMatrix<T>& lhs, //!< lhs matrix to compare
92  MappedSparseMatrix<T>& rhs, //!< rhs matrix to compare
93  const double tol) //!< comparison tolerence
94  {
95  if(
96  (lhs.GetNbrNonZeros() != rhs.GetNbrNonZeros()) ||
97  (lhs.GetNbrRows() != rhs.GetNbrRows()) ||
98  (lhs.GetNbrColumns() != rhs.GetNbrColumns())
99  )
100  {
101  return false;
102  }
103  if(utilities::is_same<dcmplx, T>::value)
104  {
105  for(auto it = lhs.m_map.begin(); it != lhs.m_map.end(); ++it)
106  {
107  uint32_t row;
108  uint32_t col;
109  utilities::Unpack2x32(it->first, row, col);
110  if(abs(rhs.Value(row,col)-lhs.Value(row,col))>tol)
111  {
112  return false;
113  }
114  }
115  }
116  else if(utilities::is_same<double,T>::value)
117  {
118  for(auto it = lhs.m_map.begin(); it != lhs.m_map.end(); ++it)
119  {
120  uint32_t row;
121  uint32_t col;
122  utilities::Unpack2x32(it->first, row, col);
123  if(abs(rhs.Value(row,col)-lhs.Value(row,col))>tol)
124  {
125  return false;
126  }
127  }
128  }
129  return true;
130  }

265  //////////////////////////////////////////////////////////////////////////////////
266  //! \brief Initialize from a CRS matrix. The data values must be the same type T
267  //!
268  //! NOTE 1: This initialization function will dealocate the original
269  //! CrsSparseMatrix data structure in the process.
270  //////////////////////////////////////////////////////////////////////////////////
271  void Initialize(
272  CrsSparseMatrix<T>* other, //!< CRS format matrix to construct from

274  const int nbrCols) //!< Specify number of matrix columns
275  {
276  m_nbrRows = nbrRows;
277  m_nbrCols = nbrCols;
278  for(int i=0; i<other->m_rowStarts.size()-1; ++i)
