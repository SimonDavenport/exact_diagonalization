35  //! \brief A function to test whether a given Fock state is in a given
36  //! quantum number sector.
37  //!
38  //! This function defines the representation of the Fock states in terms of

40  ////////////////////////////////////////////////////////////////////////////////
41  inline bool TestAngularMomentumSector(
42  const fock_t state, //!< Fock state to be tested
43  const mState_t lzTot, //!< Given angular momentum sector
44  //! to test against
45  const mState_t max) //!< Bounding angular momentum quantum number
46  {
47  // This function assumes that the Fock states are labelled by
48  // angular momentum quantum numbers from -m to m in steps of 2
49  fock_t tempState = state;
50  // Iterate over occupied states to find the total angular momentum value
51  mState_t cumulativeLz = 0;
52  while(0 != tempState)
53  {
54  // Isolate right-most bit and determine its position
55  fock_t rightMost = tempState & -tempState;
56  cumulativeLz += 2*utilities::binary::Base2Log(rightMost) - max;
57  // Mask off the current right-most bit and iterate
58  tempState = tempState ^ rightMost;
59  }
60  // Compare with the actual lzTot sector
61  return (lzTot == cumulativeLz);
62  }
