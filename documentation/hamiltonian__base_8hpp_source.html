306  ////////////////////////////////////////////////////////////////////////////////
307  //! \brief Combine partial Hamiltonians stored on each separate node into
308  //! a single combined Hamiltonian
309  //!
310  //! NOTE: There is no merge function for a hash map, so this type
311  //! will first be converted to a CRS type
312  //!
313  //! This function MUST be called on all nodes if run in parallel
314  ////////////////////////////////////////////////////////////////////////////////
315  void MergeDistributedHamiltonian(
316  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
317  {
318  if(m_data.m_matrixAllocated && mpi.m_nbrProcs > 1)
319  {
320  if(0 == mpi.m_id) // FOR THE MASTER NODE
321  {
322  utilities::cout.AdditionalInfo()<<"\n\t- GATHERING DISTRIBUTED HAMILTONIAN"<<std::endl;
323  }
324  if(utilities::_SPARSE_MAPPED_ == m_data.m_storageMethod)
325  {
326  // Reset the storage method to CCS because this will allow simple MPI
327  // transfer of the underlying index and value arrays
328  this->ResetMatrixStorageMethod(utilities::_SPARSE_CRS_, _PARALLEL_, mpi);
329  }
330  // Now gather the matrix using the SPARSE CRS or DENSE format
331  if(utilities::_SPARSE_CRS_ == m_data.m_storageMethod)
332  {
333  // Merge sparse matrix allocation from each node
334  m_crsSparseMatrix.MpiGather(0, mpi);
335  }
336  else if(utilities::_DENSE_ == m_data.m_storageMethod)
337  {
338  // Merge dense matrix allocation from each node
339  if(0 == mpi.m_id) // FOR THE MASTER NODE
340  {
341  utilities::cout.DebuggingInfo()<<"\n\t- ALLOCATING "<<((fock_t)m_data.m_fockSpaceDim*m_data.m_fockSpaceDim*(sizeof(T))/(1024.0*1024.0));
342  utilities::cout.DebuggingInfo()<<" MB TO STORE BUFFER FOR MPI GATHER"<<std::endl;
343 
344  if((fock_t)(m_data.m_fockSpaceDim*m_data.m_fockSpaceDim*(sizeof(T)))>_MEMORY_QUERY_LIMIT_)
345  {
346  std::cerr<<"\n\tWARNING: EXCEEDING "<<_MEMORY_QUERY_LIMIT_/(1024*1024*1024)<<" GB";
347  std::cerr<<" MEMORY LIMIT! PRESS ANY KEY TO CONTINUE"<<std::endl;
348  getchar();
349  }
350  // Reallocate the m_fullMatrix address on the master node
351  // to store the full matrix
352  m_fullMatrix.resize(m_data.m_fockSpaceDim*m_data.m_fockSpaceDim);
353  }
354  MPI_Barrier(mpi.m_comm);
355  MPI_Status status;
356  mpi.Gather<T>(m_fullMatrix.data(), m_data.m_nodeDim*m_data.m_fockSpaceDim, m_fullMatrix.data(),
357  m_data.m_fockSpaceDim*m_data.m_fockSpaceDim, 0, mpi.m_comm, status);
358  MPI_Barrier(mpi.m_comm);
359  }
360  if(0 != mpi.m_id)
361  {
362  this->DeallocateMatrixMemory(m_data.m_storageMethod);
363  }
364  }
365  // Reset the node dimension data to describe a matrix stored on
366  // a single node only
367  m_data.SetDataDimensions(m_data.m_fockSpaceDim, 1, mpi);
368  return;
369  }

371  ////////////////////////////////////////////////////////////////////////////////
372  //! \brief Run a check to see if the matrix is diagonal. LAPACK has trouble
373  //! processing diagonal matrices, but then if the matrix is already
374  //! diagonal we know what its eigenspectrum is anyway.
375  //!
376  //! This function MUST be called on all nodes if run in parallel
377  ////////////////////////////////////////////////////////////////////////////////
378  bool MatrixDiagonalCheck(
379  const parallelFlag_t flag, //!< Flag to set parallel implementation or not
380  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
381  {
382  int falseCount = 0;
383  int totalFalseCount = 0;
384  if(utilities::_DENSE_ == m_data.m_storageMethod && _SERIAL_ == flag)
385  {
386  for(fock_t i=0; i<m_data.m_fockSpaceDim; ++i)
387  {
388  for(fock_t j=i+1; j<m_data.m_fockSpaceDim; ++j)
389  {
390  if(utilities::is_same<T,dcmplx>::value)
391  {
392  if(abs(m_fullMatrix[i*m_data.m_fockSpaceDim+j])>_SPARSE_ZERO_TOL_)
393  {
394  return false;
395  }
396  }
397  else if(utilities::is_same<T, double>::value)
398  {
399  if(fabs(m_fullMatrix[i*m_data.m_fockSpaceDim+j])>_SPARSE_ZERO_TOL_)
400  {
401  return false;
402  }
403  }
404  }
405  }
406  return true;
407  }
408  else if(utilities::_SPARSE_CRS_ == m_data.m_storageMethod && _PARALLEL_ == flag)
409  {
410  // Determine the starting matrix index on each node
411  mpi.DivideTasks(mpi.m_id, m_data.m_fockSpaceDim, mpi.m_nbrProcs,
412  &mpi.m_firstTask, &mpi.m_lastTask, false);
413  // Iterate over all off diagonal sparse matrix elements
414  int* p_rowStarts = m_crsSparseMatrix.GetRowStarts();
415  int* p_cols = m_crsSparseMatrix.GetCols();
416  T* p_data = m_crsSparseMatrix.GetData();
417  for(int i=0; i<m_crsSparseMatrix.GetNbrRows(); ++i)
418  {
419  for(int j = p_rowStarts[i]; j<p_rowStarts[i+1]; ++j)
420  {
421  if(utilities::is_same<T,dcmplx>::value)
422  {
423  if((p_cols[j]!=(int)mpi.m_firstTask+i) && abs(p_data[j])>_SPARSE_ZERO_TOL_)
424  {
425  falseCount = 1;
426  goto end;
427  }
428  }
429  else if(utilities::is_same<T,double>::value)
430  {
431  if((p_cols[j]!=(int)mpi.m_firstTask+i) && fabs(p_data[j])>_SPARSE_ZERO_TOL_)
432  {
433  falseCount = 1;
434  goto end;
435  }
436  }
437  }
438  }
439  }
440  else
441  {
442  return false;
443  }
444  end:
445  // Count the number of false values set on all nodes
446  MPI_Allreduce(&falseCount, &totalFalseCount, 1, mpi.GetType<int>(), MPI_SUM, mpi.m_comm);
447  // The matrix is diagonal if there were no false returns from any node
448  return totalFalseCount == 0;
449  }

625  ////////////////////////////////////////////////////////////////////////////////
626  //! \brief Call a dense matrix diagonalization routine from LAPACK
627  //!
628  //! This function MUST be called on all nodes if run in parallel
629  //!
630  //! NOTE: deallocates matrix storage memory once completed
631  //!
632  //! \return true if success, false if failure
633  ////////////////////////////////////////////////////////////////////////////////
634  bool FullDiagonalize(
635  iSize_t& nbrEigenvalues, //!< Number of eigenvalues to find
636  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
637  {
638  MPI_Barrier(mpi.m_comm);
639  if(0 == mpi.m_id) // FOR THE MASTER NODE
640  {
641  utilities::cout.SecondaryOutput()<<"\n\t============ DIAGONALIZING HAMILTONIAN WITH LAPACK ============ "<<std::endl;
642  }
643  // Make sure that all nodes know how many eigenvalues we are including
644  this->SetNbrEigenvalues(nbrEigenvalues, mpi);
645  // LAPACK will only work on a single node
646  // (use SCALAPACK for parallel algorithms?)
647  this->MergeDistributedHamiltonian(mpi);
648  if(utilities::_SPARSE_MAPPED_ == m_data.m_storageMethod || utilities::_SPARSE_CRS_ == m_data.m_storageMethod)
649  {
650  // Need to convert to dense storage format first
651  this->ResetMatrixStorageMethod(utilities::_DENSE_, _SERIAL_, mpi);
652  }
653  // Proceed on the master node
654  if(0 == mpi.m_id) // FOR THE MASTER NODE
655  {
656  if(m_data.m_matrixAllocated)
657  {
658  m_eigenvalues.resize(m_data.m_nbrEigenvalues);
659  m_eigenvectors.resize(m_data.m_nbrEigenvalues*m_data.m_fockSpaceDim);
660  if(this->MatrixDiagonalCheck(_SERIAL_,mpi))
661  {
662  utilities::cout.SecondaryOutput()<<"\n\tMATRIX WAS ALREADY DIAGONAL - GENERATING SPECTRUM DIRECTLY"<<std::endl;
663  // For a diagonal matrix, the eigenspectrum can be easily extracted
664  double* diagBuffer = new double[m_data.m_fockSpaceDim];
665  double* sortBuffer = new double[m_data.m_fockSpaceDim];
666  // Populate a buffer with diagonal entries
667  for(fock_t i=0; i<m_data.m_fockSpaceDim; ++i)
668  {
669  diagBuffer[i] = std::real(m_fullMatrix[i*m_data.m_fockSpaceDim+i]);
670  }
671  memcpy(sortBuffer, diagBuffer, m_data.m_fockSpaceDim*sizeof(double));
672  // Put the values in descending order
673  std::sort(sortBuffer,sortBuffer+m_data.m_fockSpaceDim);
674  // Read out the lowest eigenvalues
675  // and generate their corresponding eigenvectors
676  // by generating zero vectors and specifying
677  // a 1.0 value for each degenerate value,
678  // the normalizing at the end
679  for(iSize_t i=0; i<m_data.m_nbrEigenvalues; ++i)
680  {
681  m_eigenvalues[i] = sortBuffer[i];
682  double norm = 0.0;
683  for(fock_t j=0; j<m_data.m_fockSpaceDim; ++j)
684  {
685  if(fabs(m_eigenvalues[i]-diagBuffer[j])<_SPARSE_ZERO_TOL_)
686  {
687  m_eigenvectors[i*m_data.m_fockSpaceDim+j] = 1.0;
688  norm += 1.0;
689  }
690  else
691  {
692  m_eigenvectors[i*m_data.m_fockSpaceDim+j] = 0.0;
693  }
694  }
695  for(fock_t j=0; j<m_data.m_fockSpaceDim; ++j)
696  {
697  m_eigenvectors[i*m_data.m_fockSpaceDim+j] /= norm;
698  }
699  }
700  delete[] diagBuffer;
701  delete[] sortBuffer;
702  }
703  else
704  {
705  utilities::cout.SecondaryOutput()<<"\n\tCALLING LAPACK ROUTINE"<<std::endl;
706  utilities::Timer timer; // Time the operation
707  timer.Start();
708  utilities::linearAlgebra::DiagonalizeSymmetricMatrix<T>(m_fullMatrix.data(), m_eigenvectors.data(),
709  m_eigenvalues.data(), m_data.m_fockSpaceDim,
710  m_data.m_nbrEigenvalues, 'U');
711  timer.Stop();
712  }
713  // Now deallocate the memory assigned to store the Hamiltonian
714  this->DeallocateMatrixMemory(m_data.m_storageMethod);
715  utilities::cout.AdditionalInfo()<<"\n\t============ HAMILTONIAN SUCESSFULLY DIAGONALIZED ============ "<<std::endl;
716  m_data.m_eigenvaluesCalculated = true;
717  m_data.m_eigenvectorsCalculated = true;
718  this->PrintEigensystem();
719  }
720  else
721  {
722  m_data.m_eigenvaluesCalculated = false;
723  m_data.m_eigenvectorsCalculated = false;
724  }
725  }
726  mpi.Sync(&m_data.m_eigenvaluesCalculated, 1, 0);
727  mpi.Sync(&m_data.m_eigenvectorsCalculated, 1, 0);
728  // We need to scatter the eigenvectors over multiple nodes if we want to perform
729  // any further data analysis in parallel
730  if(m_data.m_eigenvaluesCalculated)
731  {
732  // Synchronize with the master node
733  mpi.Sync(&m_eigenvalues, 0);
734  }
735  if(m_data.m_eigenvectorsCalculated)
736  {
737  // Call MPI scatter on the eigenvectors
738  // First copy the existing eigenvectors into the scatter buffer
739  T* eigenvectorsBuffer = 0;
740  if(0 == mpi.m_id) // FOR THE MASTER NODE
741  {
742  eigenvectorsBuffer = new T[m_data.m_nbrEigenvalues*m_data.m_fockSpaceDim];
743  memcpy(eigenvectorsBuffer, m_eigenvectors.data(), m_data.m_nbrEigenvalues*m_data.m_fockSpaceDim*sizeof(T));
744  }
745  mpi.DivideTasks(mpi.m_id, m_data.m_fockSpaceDim, mpi.m_nbrProcs, &mpi.m_firstTask, &mpi.m_lastTask, false);
746  fock_t eigenvectorNodeDim = mpi.m_lastTask - mpi.m_firstTask + 1;
747  m_eigenvectors.resize(m_data.m_nbrEigenvalues*eigenvectorNodeDim);
748  // MPI scatter each eigenvector
749  for(iSize_t e=0; e<m_data.m_nbrEigenvalues; ++e)
750  {
751  MPI_Status status;
752  mpi.Scatter<T>(eigenvectorsBuffer+e*m_data.m_fockSpaceDim, m_data.m_fockSpaceDim,
753  m_eigenvectors.data()+e*eigenvectorNodeDim, eigenvectorNodeDim, 0, mpi.m_comm, status);
754  }
755  if(0!=eigenvectorsBuffer) delete[] eigenvectorsBuffer;
756  }
757  return m_data.m_eigenvaluesCalculated && m_data.m_eigenvectorsCalculated;
758  }

760  ////////////////////////////////////////////////////////////////////////////////
761  //! \brief Call a sparse Lanczos algorithm from the ARPACK/PARPACK library
762  //!
763  //! This function MUST be called on all nodes if run in parallel
764  //!
765  //! NOTE: deallocates matrix storage memory once completed
766  //!
767  //! \return true if success, false if failure
768  ////////////////////////////////////////////////////////////////////////////////
769  template <class L1, class L2>
770  bool LanczosDiagonalize(
771  L1* quadraticLookUpTables, //!< Class container for look-up tables
772  L2* quarticLookUpTables, //!< Class container for look-up tables
773  MatrixVectorFunction<T, L1, L2>& mv,//!< Class to perform matrix-vector operation
774  iSize_t& nbrEigenvalues, //!< Number of eigenvalues to find
775  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
776  {
777  ////////////////////////////////////////////////////////////////////////////////
778  #if _ENABLE_ARPACK_
779  MPI_Barrier(mpi.m_comm);
780  if(0 == mpi.m_id) // FOR THE MASTER NODE
781  {
782  utilities::cout.SecondaryOutput()<<"\n\t============ DIAGONALIZING HAMILTONIAN WITH ARPACK ============ "<<std::endl;
783  }
784  // Make sure that all nodes know how many eigenvalues we are including
785  this->SetNbrEigenvalues(nbrEigenvalues,mpi);
786  if(utilities::_DENSE_ == m_data.m_storageMethod || utilities::_SPARSE_MAPPED_ == m_data.m_storageMethod)
787  {
788  // Need to convert to sparse-ccs storage format first
789  // for fast matrix-vector operations and for the
790  // MergeDistributedHamiltonian function
791  if(0 == mpi.m_id) // FOR THE MASTER NODE
792  {
793  utilities::cout.SecondaryOutput()<<"\n\t- RESETTING STORAGE METHOD TO CRS"<<std::endl;
794  }
795  this->ResetMatrixStorageMethod(utilities::_SPARSE_CRS_, _PARALLEL_, mpi);
796  }
797  if(m_data.m_matrixAllocated)
798  {
799  // Allocate memory to store eigenvalues/eigenvectors
800  // Distributed over a number of different nodes
801  m_eigenvalues.resize(m_data.m_nbrEigenvalues);
802  m_eigenvectors.resize(m_data.m_nbrEigenvalues*m_data.m_nodeDim);
803  if(this->MatrixDiagonalCheck(_PARALLEL_, mpi))
804  {
805  if(0 == mpi.m_id) // For the master node
806  {
807  utilities::cout.SecondaryOutput()<<"\n\tMATRIX WAS ALREADY DIAGONAL - GENERATING SPECTRUM DIRECTLY"<<std::endl;
808  }
809  // For a diagonal matrix, the eigenspectrum can be easily extracted
810  double* diagBuffer = new double[m_data.m_nodeDim];
811  // Populate a buffer with diagonal entries
812  mpi.DivideTasks(mpi.m_id, m_data.m_fockSpaceDim, mpi.m_nbrProcs, &mpi.m_firstTask, &mpi.m_lastTask, false);
813  for(fock_t i=mpi.m_firstTask; i<=mpi.m_lastTask; ++i)
814  {
815  diagBuffer[i-mpi.m_firstTask] = std::real(m_crsSparseMatrix.Value(i-mpi.m_firstTask, i));
816  }
817  // Merge the diagonal elements on the master node
818  double* mergeBuffer=0;
819  if(0 == mpi.m_id) // For the master node
820  {
821  mergeBuffer = new double[m_data.m_fockSpaceDim];
822  }
823  MPI_Status status;
824  mpi.Gather<double>(diagBuffer, m_data.m_nodeDim, mergeBuffer, m_data.m_fockSpaceDim, 0, mpi.m_comm, status);
825  if(0 == mpi.m_id) // proceed on the master node only
826  {
827  double* sortBuffer = new double[m_data.m_fockSpaceDim];
828  memcpy(sortBuffer, mergeBuffer, m_data.m_fockSpaceDim*sizeof(double));
829  // Put the values in descending order
830  std::sort(sortBuffer, sortBuffer+m_data.m_fockSpaceDim);
831  // Read out the lowest eigenvalues
832  // and generate their corresponding eigenvectors
833  // by generating zero vectors and specifying
834  // a 1.0 value for each degenerate value,
835  // the normalizing at the end
836  for(iSize_t i=0; i<m_data.m_nbrEigenvalues; ++i)
837  {
838  m_eigenvalues[i] = sortBuffer[i];
839  }
840  delete[] sortBuffer;
841  delete[] mergeBuffer;
842  }
843  // Set the eigenvalues on all nodes
844  mpi.Sync(&m_eigenvalues, 0);
845  // Set the eigenvectors
846  for(iSize_t i=0; i<m_data.m_nbrEigenvalues; ++i)
847  {
848  double norm = 0.0;

850  for(fock_t j=mpi.m_firstTask; j<=mpi.m_lastTask; ++j)

856  }
857  else
858  {
859  m_eigenvectors[i*m_data.m_nodeDim+j-mpi.m_firstTask] = 0.0;
860  }
861  }

864  for(fock_t j=mpi.m_firstTask; j<=mpi.m_lastTask; ++j)
865  {
866  m_eigenvectors[i*m_data.m_nodeDim+j-mpi.m_firstTask] /= totalNorm;
867  }
868  }
869  MPI_Barrier(mpi.m_comm);
870  // Clear buffer allocation
871  delete[] diagBuffer;
872  }

881  {
882  std::cerr<<"\n\tWARNING - CALLING (P)ARPACK FOR A VERY SMALL MATRIX MAY LEAD TO AN ERROR. USE LAPACK INSTEAD?"<<std::endl;
883  }
884  }
885  utilities::Timer timer;
886  if(0 == mpi.m_id) // FOR THE MASTER NODE

888  timer.Start(); // Time the operation
889  }

893  &(mv.inputVectorBuffer), &(mv.outputVectorBuffer), mpi);

895  m_arpackWrapper.DiagonalizeSymmetricMatrix<T>(std::bind(&MatrixVectorFunction<T, L1, L2>::Simple,
896  mv, std::placeholders::_1, std::placeholders::_2,
897  std::placeholders::_3), m_eigenvectors.data(),
898  m_eigenvalues.data(), m_data.m_nodeDim, m_data.m_fockSpaceDim,
899  m_data.m_nbrEigenvalues, mpi);
900  if(0 == mpi.m_id) // FOR THE MASTER NODE
901  {
902  timer.Stop();
903  }
904  }
905  this->DeallocateMatrixMemory(m_data.m_storageMethod);
906  MPI_Barrier(mpi.m_comm);
907  if(0 == mpi.m_id)
908  {
909  utilities::cout.AdditionalInfo()<<"\n\t============ HAMILTONIAN SUCESSFULLY DIAGONALIZED ============ "<<std::endl;
910  }
911  m_data.m_eigenvaluesCalculated = true;
912  m_data.m_eigenvectorsCalculated = true;
913  if(0 == mpi.m_id)
914  {
915  this->PrintEigensystem();
916  }
917  }
918  else
919  {
920  // ERROR
921  m_data.m_eigenvaluesCalculated = false;
922  m_data.m_eigenvectorsCalculated = false;
923  }
924  return m_data.m_eigenvaluesCalculated && m_data.m_eigenvectorsCalculated;

927  // diagonalization failed

929  #endif
