96  //!
97  void BaseGenerate4KTable(

141  ////////////////////////////////////////////////////////////////////////////////
142  //! \brief Convert two-body pseudopotential coefficients into a list of
143  //! terms of the form Vkkkk.
144  //!
145  //! The coefficients are given by:
146  //!
147  //! \[ V_{k1,k2,k3,k4} = \delta (k3+k4-k1-k2) * sum_{L=0^m_maxLz} sum_{M=-L^L}
148  //! V_L <L,M||k1>|k2> <k4|<k3||L,M>
149  ////////////////////////////////////////////////////////////////////////////////
150  void BaseTermsFromPseudopotentials(
151  std::vector<double>* listOfQuarticTerms,
152  //!< List of coefficients to populate
153  const iSize_t llIndex, //!< Landau level index
154  const utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
155  const
156  {
157  if(0 == mpi.m_id) // FOR THE MASTER NODE
158  {
159  utilities::cout.AdditionalInfo()<<"\n\t- GENERATING TERM COEFFICIENTS FROM TWO-BODY PSEUDOPOTENTIALS for LL "<<llIndex<<std::endl;
160  }
161  mState_t maxLz;
162  std::vector<double> pseudopotentials;
163  if(llIndex == 0)
164  {
165  maxLz = m_params.m_maxLz;
166  pseudopotentials = m_params.m_twoBodyPseudopotentials;
167  }
168  else if(llIndex == 1)
169  {
170  maxLz = m_params.m_maxLz2;
171  pseudopotentials = m_params.m_twoBodyPseudopotentials2LL;
172  }
173  else
174  {
175  if(0 == mpi.m_id) // FOR THE MASTER NODE
176  {
177  utilities::cout.AdditionalInfo()<<"\n\tERROR in GetTermsFromPseudopotentials: Invalid ll index "<<llIndex<<std::endl;
178  }
179 
180  exit(EXIT_FAILURE);
181  }
182  auto it_table = listOfQuarticTerms->begin();
183  // Generate a look-up table of Clebsch Gordan coefficients
184  utilities::ClebschGordanCoefficients<double> clebschGordan;
185  clebschGordan.AddMultiplet(maxLz, maxLz);
186  if(0 == mpi.m_id) // FOR THE MASTER NODE
187  {
188  utilities::cout.AdditionalInfo()<<"\n\t- GENERATED TEMPORARY TABLE OF "<<clebschGordan.GetSize()<<" CLEBSCH GORDAN COEFFICIENTS."<<std::endl;
189  }
190  for(mState_t m4=maxLz; m4>=-maxLz; m4-=2)
191  {
192  for(mState_t m3=maxLz; m3>=-maxLz; m3-=2)
193  {
194  for(mState_t m2=maxLz; m2>=-maxLz; m2-=2, ++it_table)
195  {
196  if(it_table != listOfQuarticTerms->end())
197  {
198  double total = 0.0;
199  mState_t m1 = this->FindM1(m2, m3, m4, maxLz);
200  if(m1 <= maxLz)
201  {
202  // Perform sum over L and M
203  for(mState_t L=0; L<2*pseudopotentials.size(); L+=2)
204  {
205  double temp = 0.0;
206  // For consistency with known results, the L value must
207  // be reflected in this way:
208  mState_t Lact = 2*maxLz-L;
209  mState_t M = m1+m2;
210 
211  if(-Lact<=M && M<=Lact)
212  {
213  // Add terms of the form
214  // <L,M||m1>|m2> <m4|<m3||L,M>
215  temp += clebschGordan.Value(maxLz,m1,maxLz,m2,Lact,M)*clebschGordan.Value(maxLz,m3,maxLz,m4,Lact,M);
216  }
217  if(L/2<pseudopotentials.size())
218  {
219  total -= pseudopotentials[L/2]*temp;
220  }
221  }
222  }
223  *it_table = total;
224  }
225  }
226  }
227  }
228  return;
229  }

231  //!
232  //! Diagonalize our Hamiltonian using a selected method
233  //!
234  template <class L1, class L2>
235  void BaseDiagonalize(
236  L1* quadraticLookUpTables, //!< Class container for look-up tables
237  L2* quarticLookUpTables, //!< Class container for look-up tables
238  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
239  {
240  if(m_params.m_hamiltonianBuilt)
241  {
242  ////// CALL SELECTED DIAGONALIZATION ROUTINE
243  // Switch to LAPACK for very small matrices
244  if(m_hamiltonian.m_data.m_fockSpaceDim<60)
245  {
246  if(0 == mpi.m_id)
247  {
248  utilities::cout.SecondaryOutput()<<"\n\tMATRIX DIMENSION <60 - SETTING DIAGONALIZATION METHOD TO FULL\n"<<std::endl;
249  }
250  m_params.m_method = _FULL_;
251  }
252  if(_FULL_ == m_params.m_method)
253  {
254  m_params.m_hamiltonianDiagonalized = m_hamiltonian.FullDiagonalize(m_params.m_nbrEigenvaluesToFind,mpi);
255  }
256  else if(_LANCZOS_ == m_params.m_method)
257  {
258  // Update the file names for the input/output ARPACK vectors
259  std::stringstream inFileName;
260  std::stringstream outFileName;
261  inFileName.str("");
262  outFileName.str("");
263  inFileName<<m_params.m_inPath<<"/"<<m_params.m_initialVectorFile;
264  outFileName<<m_params.m_outPath<<"/"<<m_params.m_finalVectorFile;
265  if(m_params.m_blockDiagonalize)
266  {
267  inFileName<<"_sector_"<<m_params.m_totalLz;
268  outFileName<<"_sector_"<<m_params.m_totalLz;
269  }
270  inFileName<<"_proc_"<<mpi.m_id<<".bin";
271  outFileName<<"_proc_"<<mpi.m_id<<".bin";
272  m_hamiltonian.UpdateArpackFileNames(inFileName.str(), outFileName.str(), mpi);
273  MatrixVectorFunction<double, L1, L2> mv(m_hamiltonian, mpi);
274  m_params.m_hamiltonianDiagonalized = m_hamiltonian.LanczosDiagonalize(
275  quadraticLookUpTables,quarticLookUpTables, mv, m_params.m_nbrEigenvaluesToFind, mpi);
276  }
277  if(!m_params.m_hamiltonianDiagonalized)
278  {
279  if(0 == mpi.m_id) // FOR THE MASTER NODE
280  {
281  std::cerr<<"\n\tWARNING - NOT ABLE TO RUN DIAGONALIZATION ROUTINE."<<std::endl;
282  }
283  }
284  }
