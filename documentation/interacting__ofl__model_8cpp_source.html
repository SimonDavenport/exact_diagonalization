107  //! \brief Generate tables describing the quadratic and quartic terms in the
108  //! Hamiltonian
109  //!
110  //! This implements the Hamiltonian given e.g. in eq (5) of PRL 109,265301
111  ////////////////////////////////////////////////////////////////////////////////
112  void InteractingOflModel::BuildTermTables(
113  boost::program_options::variables_map* optionList,
114  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
115  {
116  ////// POPULATE LOOK-UP TABLES FOR THE QUADRATIC AND QUARTIC TERM //////////
117  ///// COEFFICIENTS OF OUR HAMILTONIAN //////////
118  if(0 == mpi.m_id) // FOR THE MASTER NODE
119  {
120  utilities::cout.MainOutput()<<"\n\t============ BUILDING MATRIX ELEMENT LOOK-UP TABLES ============"<<std::endl;
121  }
122  double matrixElementTol;
123  iSize_t minBlochCutOff;
124  iSize_t maxBlochCutOff;
125  iSize_t cutOffStep;
126  if(0 == mpi.m_id) // FOR THE MASTER NODE
127  {
128  GetOption(optionList, matrixElementTol, "tol-vkkkk", _AT_, mpi);
129  GetOption(optionList, minBlochCutOff, "min-k-cut", _AT_, mpi);
130  GetOption(optionList, maxBlochCutOff, "max-k-cut", _AT_, mpi);
131  GetOption(optionList, cutOffStep, "k-cut-step", _AT_, mpi);
132  }
133  mpi.ExitFlagTest();
134  mpi.Sync(&matrixElementTol, 1, 0);
135  mpi.Sync(&minBlochCutOff, 1, 0);
136  mpi.Sync(&maxBlochCutOff, 1, 0);
137  mpi.Sync(&cutOffStep, 1, 0);
138  // Calculate the Vkkkk coefficients iteratively, increasing the value of the
139  // cut off in the Bloch coefficient calculation until we converge on a
140  // stable result to within some tolerance level
141  // we also use these solutions to populate the quadratic term table
142  m_quarticTables.Initialize(m_params.m_dimX*m_params.m_dimY, mpi);
143  std::vector<int> gxTable(m_quarticTables.GetDimension());
144  std::vector<int> gyTable(m_quarticTables.GetDimension());
145  this->Generate4KTable(m_quarticTables.GetKTable(), &gxTable, &gyTable, mpi);
146  // Allocate memory to store a temporary look-up table of quartic terms
147  // but only store on the master node
148  std::vector<dcmplx> listOfQuarticTerms;
149  if(0 == mpi.m_id) // FOR THE MASTER NODE
150  {
151  listOfQuarticTerms.resize(m_quarticTables.GetDimension());
152  }
153  // Allocate memory to store a list of Bloch wave function tables
154  NonInteractingOflModel* blochWaveFunctionTable = new NonInteractingOflModel[m_params.m_dimX*m_params.m_dimY];
155  iSize_t cutOff = minBlochCutOff;
156  double currentTol = 1000.0;
157  double prevTol = currentTol;
158  // Generate the first approximation
159  NonInteractingOflModelGrid modelGrid;
160  modelGrid.GenerateBlochWaveFunctionTable(cutOff, m_params.m_dimX, m_params.m_dimY, m_params.m_offsetX,
161  m_params.m_offsetY, blochWaveFunctionTable, &m_oflParameters, mpi);
162  this->GenerateQuarticTerms(m_quarticTables.GetVTable(), m_quarticTables.GetDimension(), blochWaveFunctionTable,
163  &gxTable, &gyTable, mpi);
164  do
165  {
166  if(0 == mpi.m_id) // FOR THE MASTER NODE
167  {
168  if(cutOff>maxBlochCutOff)
169  {
170  std::cerr<<"WARNING in BuildTermTables: max Bloch cut off reached without converging!"<<std::endl;
171  break;
172  }
173  utilities::cout.DebuggingInfo()<<"\n\t- PERFORMING MORE ACCURATE CALCULATION "<<std::endl;
174  }
175  mpi.ExitFlagTest();
176  // Increase the range of Bloch coefficients used in the calculation
177  cutOff += cutOffStep;
178  // Re-generate the coefficient table
179  modelGrid.GenerateBlochWaveFunctionTable(cutOff, m_params.m_dimX, m_params.m_dimY, m_params.m_offsetX,
180  m_params.m_offsetY, blochWaveFunctionTable, &m_oflParameters, mpi);
181  this->GenerateQuarticTerms(&listOfQuarticTerms, m_quarticTables.GetDimension(), blochWaveFunctionTable,
182  &gxTable, &gyTable, mpi);
183  // Get the maximum abs difference in Vkkkk values
184  if(0 == mpi.m_id) // FOR THE MASTER NODE
185  {
186  auto it_list1 = listOfQuarticTerms.begin();
187  auto it_list2 = m_quarticTables.GetVTable()->begin();
188  currentTol = 0.0;
189  for(iSize_t i=0; i<m_quarticTables.GetDimension(); ++i, ++it_list1, ++it_list2)
190  {
191  const double temp = fabs(abs(*it_list1)-abs(*it_list2));
192  if(temp>currentTol)
193  {
194  currentTol = temp;
195  }
196  }
197  *(m_quarticTables.GetVTable()) = listOfQuarticTerms;
198  utilities::cout.SecondaryOutput()<<"\n\tBLOCH BAND CUT OFF: "<<cutOff<<"\tMAX CHANGE IN |Vkkkk|: "<<currentTol<<std::endl;
199  }
200  mpi.Sync(&currentTol, 1, 0);
201  // Break out if the previous tolerance was better than the current
202  // tolerance (a precision limit has been reached)
203  if(fabs(prevTol)<fabs(currentTol))
204  {
205  std::cerr<<"\n\tWARNING: DUE TO NON CONVERGENCE, Vkkkk CALCULATION EXITED BEFORE SPECIFIED TOLERENCE REACHED!"<<std::endl;
206  break;
207  }
208  else
209  {
210  prevTol = currentTol;
211  }
212  }
213  while(currentTol>matrixElementTol);
214  // Generate the corresponding table of quadratic terms
215  m_quadraticTables.Initialize(m_params.m_dimX*m_params.m_dimY,mpi);
216  if(0 == mpi.m_id) // FOR THE MASTER NODE
217  {
218  utilities::cout.AdditionalInfo()<<"\n\t- MAXIMUM TOLERENCE ("<<std::max(matrixElementTol,fabs(currentTol))<<") REACHED!"<<std::endl;
219  auto it_quadraticTable = m_quadraticTables.GetVTable()->begin();
220  for(iSize_t i=0; i<m_quadraticTables.GetDimension(); ++i, ++it_quadraticTable)
221  {
222  *(it_quadraticTable) = blochWaveFunctionTable[i].GetBandEnergy(0);
223  }
224  }
225  // Multiply all quartic table values by a constant coefficient
226  if(0 == mpi.m_id) // FOR THE MASTER NODE
227  {
228  m_quarticTables.SetCoefficient(0.5*m_params.m_interactionStrength);
229  }
230  // Synchronize with the master node
231  if(0 == mpi.m_id) // FOR THE MASTER NODE
232  {
233  utilities::cout.AdditionalInfo()<<"\n\t- MPI SYNC HAMILTONIAN COEFFICIENT TABLES"<<std::endl;
234  }
235  m_quadraticTables.MpiSynchronize(0, mpi);
236  m_quarticTables.MpiSynchronize(0, mpi);
237  // Calculate sigmaZ map
238  this->CalcualteSigmaZMap(blochWaveFunctionTable, mpi);
239  // Convert to Wannier basis if requested
240  if(m_params.m_useWannierBasis)
241  {
242  this->ChangeToWannierBasis(blochWaveFunctionTable, matrixElementTol, mpi);
243  }
244  // g Tables and blochWaveFunctionTable no longer required
245  delete[] blochWaveFunctionTable;
246  if(_HASH_ == m_params.m_setTableFormat)
247  {
248  this->ConvertTableFormat(mpi);
249  }
250  m_params.m_termTablesBuilt = true;
251  return;
252  }

255  //! \brief Generate a list of the unique k1 value corresponding to a given k2,k3
256  //! and k4 assuming 2D momentum conservation.
257  ////////////////////////////////////////////////////////////////////////////////
258  void InteractingOflModel::Generate4KTable(
259  std::vector<kState_t>* kTable, //!< Preallocated table of k-values
260  std::vector<int>* gxTable, //!< Preallocated table of reciprocal lattice values
261  std::vector<int>* gyTable, //!< Preallocated table of reciprocal lattice values
262  const utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
263  {
264  if(0 == mpi.m_id) // FOR THE MASTER NODE
265  {
266  utilities::cout.AdditionalInfo()<<"\n\t- GENERATING MOMENTUM CONSERVING k1,k2,k3,k4 TABLE"<<std::endl;
267  }
268  kState_t kMax = m_params.m_dimX*m_params.m_dimY;
269  auto it_kTable = kTable->begin();
270  auto it_gxTable = gxTable->begin();
271  auto it_gyTable = gyTable->begin();
272  for(kState_t k4=0; k4<kMax; ++k4)
273  {
274  for(kState_t k3=k4; k3<kMax; ++k3)
275  {
276  for(kState_t k2=0; k2<kMax; ++k2, ++it_kTable, ++it_gxTable, ++it_gyTable)
277  {
278  if(it_kTable < kTable->end() && it_gxTable < gxTable->end() && it_gyTable < gyTable->end())
279  {
280  // For every k2,k3,k4 there is a unique k1 in the BZ
281  // such that k1+k2-k3-k4 = a reciprocal lattice vector
282  // as G = x GX + y GY
283  int x,y;
284  kState_t k1 = this->FindK1(k2, k3, k4, x, y);
285  *(it_kTable) = k1;
286  *(it_gxTable) = x;
287  *(it_gyTable) = y;
288  }
289  }
290  }
291  }
292  MPI_Barrier(mpi.m_comm);
293  return;
294  }

438  ////////////////////////////////////////////////////////////////////////////////
439  //! \brief Generate the coefficients V_{k1,k2,k3,k4} in the interacting Hamiltonian
440  //!
441  //! The coefficients are given by:
442  //!
443  //! \[ V_{k1,k2,k3,k4} = \sum_\sigma \sum_{G_A,G_B,G_C,G_D} \delta (k3+k4-k1-k2+G_C+G_D-G_A-G_B)
444  //! * c^{k1,*}_{G_A,\sigma} c^{k2,*}_{G_B,\bar{\sigma}} c^k3_{G_C,\bar{\sigma}} c^k4_{G_D,\sigma} \]
445  //!
446  //! The sum over G_A,...G_D is greatly simplified by writing is as two
447  //! independent sets of convolutions for the spin-up/spin-down terms separately.
448  //!
449  //! The calculation is optimized by mapping V_{k2,k1,k4,k3} to V_{k1,k2,k3,k4}
450  //!
451  //! In pseudo-code:
452  //!
453  //! V_up = {x*,y*,z*}^{x,y,z} = {xz*,xy*+yz*,xx*+yy*+zz*,yx*+zy*,zx*}
454  //!
455  //! The the sum over G_A,..G_D is given by V_up.V_down.
456  ////////////////////////////////////////////////////////////////////////////////
457  void InteractingOflModel::GenerateQuarticTerms(
458  std::vector<dcmplx>* listOfQuarticTerms,
459  //!< A pre-declared array to contain the
460  //! look-up table

462  //! (full value must be used on all nodes)
463  NonInteractingOflModel* blochTable,
464  //!< Lookup table of Bloch wave functions
465  std::vector<int>* gxTable, //!< Momentum conserving reciprocal lattice vectors (x)
466  std::vector<int>* gyTable, //!< Momentum conserving reciprocal lattice vectors (y)
467  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
468  const
469  {

471  {
472  utilities::cout.AdditionalInfo()<<"\n\t- GENERATING QUARTIC TERMS\n"<<std::endl;
473  }
474  // Determine some dimensions used in calculating the convolutions
475  const iSize_t blochTableXDim = 2*blochTable->GetCutOffX()+1;
476  const iSize_t blochTableYDim = 2*blochTable->GetCutOffY()+1;
477  const iSize_t convolutionDim = utilities::GetArrayConvolution2DSize(blochTableXDim, blochTableYDim);
478  const iSize_t convolutionArrayOffset = utilities::GetArrayConvolution1DSize(blochTableXDim, blochTableYDim);
479  // Get the total number of k-states in the discretized Brillouin zone
480  kState_t kMax = m_params.m_dimX*m_params.m_dimY;
481  // Allocate further memory to store temporary copies of the Bloch wave function tables
482  // and their convolutions
483  dcmplx* blochAup = new dcmplx[blochTableXDim*blochTableYDim];
484  dcmplx* blochAdown = new dcmplx[blochTableXDim*blochTableYDim];
485  dcmplx* blochBup = new dcmplx[blochTableXDim*blochTableYDim];
486  dcmplx* blochBdown = new dcmplx[blochTableXDim*blochTableYDim];
487  dcmplx* blochCup = new dcmplx[blochTableXDim*blochTableYDim];
488  dcmplx* blochCdown = new dcmplx[blochTableXDim*blochTableYDim];
489  dcmplx* blochDup = new dcmplx[blochTableXDim*blochTableYDim];
490  dcmplx* blochDdown = new dcmplx[blochTableXDim*blochTableYDim];
491  dcmplx* convolution1 = new dcmplx[convolutionDim];
492  dcmplx* convolution2 = new dcmplx[convolutionDim];
493  dcmplx* convolution3 = new dcmplx[convolutionDim];
494  dcmplx* convolution4 = new dcmplx[convolutionDim];
495  // Populate the list of quadratic terms
496  mpi.DivideTasks(mpi.m_id, dimension, mpi.m_nbrProcs, &mpi.m_firstTask, &mpi.m_lastTask, false);
497  // Allocate memory to store part of the result on each node
498  iSize_t taskDim = mpi.m_lastTask - mpi.m_firstTask+1;
499  dcmplx* temp = new dcmplx[taskDim];
500  NonInteractingOflModel* blochTable4 = blochTable;
501  dcmplx* p_temp = temp;
502  iSize_t counter = 0;
503  // Display a progress bar

514  blochTable4->GetBlochCoefficients(blochDdown, 0, _SPIN_DOWN_, _NO_CONJUGATE_);

520  blochTable3->GetBlochCoefficients(blochCdown, 0, _SPIN_DOWN_, _NO_CONJUGATE_);

524  {
525  blochTable2->GetBlochCoefficients(blochBup, 0, _SPIN_UP_, _CONJUGATE_);
526  blochTable2->GetBlochCoefficients(blochBdown, 0, _SPIN_DOWN_, _CONJUGATE_);
527  // For every k2,k3,k4 there is a unique k1 in the BZ
528  // such that k1+k2-k3-k4 = a reciprocal lattice vector

543  blochTableYDim, convolution2, offset1,
544  offset2, offset3, offset4);

546  blochTableYDim, convolution3, 0, 0, 0, 0);
547  utilities::ArrayConvolution2D<dcmplx, dcmplx, sizeof(dcmplx)>(blochCup, blochAdown, blochTableXDim,
548  blochTableYDim, convolution4, offset1,
549  offset2, offset3, offset4);

551  utilities::KahanAccumulation<dcmplx> Vkkkk;

556  }

558  *p_temp = Vkkkk.m_sum;
559  ++p_temp;

561  {
562  progress.Display(counter+1);
563  }
564  }
565  }
566  }
567  }

569  {

571  }

573  delete[] blochAup;
574  delete[] blochAdown;
575  delete[] blochBup;
576  delete[] blochBdown;
577  delete[] blochCup;
578  delete[] blochCdown;
579  delete[] blochDup;
580  delete[] blochDdown;
581  delete[] convolution1;
582  delete[] convolution2;
583  delete[] convolution3;
584  delete[] convolution4;

588  delete[] temp;

691  ////////////////////////////////////////////////////////////////////////////////
692  //! \brief Implement a change of basis to a cylindrical geometry using the basis
693  //! of hybrid localized Wannier states as defined in Eq 32 of PRB 86,085129 (2012)
694  //!
695  //! This function updates the quartic (Vkkkk) and quadratic e_kk terms to a form
696  //! appropriate for the new basis (note that the quadratic term is no longer
697  //! diagonal in the new basis).
698  //!
699  //! The change of basis can be written as:
700  //!
701  //! |x,k_y> = exp(-2*I*PI*offsetX x/dimX) exp^{Iphi_y(x,ky)}/sqrt(dimX)
702  //! sum_kx exp(-2*I*PI*kx x/dimX) {[[lambda_x(kx)]^kx]/product_{kappa=0}^{kx}
703  //! A_x (kappa,ky)} |kx,ky>
704  //!
705  //! The change of basis can be encoded in a matrix:
706  //!
707  //! c^+_{x,ky} = sum_{kx,ky} U_{(x,ky),(kx,ky)} c^+_{kx,ky}
708  //!
709  //! OR
710  //!
711  //! c^+_{kx,ky} = sum_{x,ky} U^-1_{(kx,ky),(x,ky)} c^+_{x,ky}
712  //!
713  //! where U is unitary and diagonal in ky. Then
714  //!
715  //! V_{(x,ky)_1,...,(x,ky)_4} = sum_{(kx,ky)_1,...,(kx,ky)_4)} V_{(kx,ky)_1,...,(kx,ky)_4}
716  //! * U^-1_{(kx,ky)_1,(x,ky)_1} U^-1_{(kx,ky)_2,(x,ky)_2}
717  //! * U_{(x,ky)_3,(kx,ky)_3} U_{(x,ky)_4,(kx,ky)_4}
718  //!
719  //! The we substitute this change of basis into the existing Hamiltonian
720  //! and calculate the new coefficients
721  //!
722  //! Since we no longer conserve kx momentum, the coefficient look-up tables
723  //! are now more conveniently stored as a multi-key hash multi-maps
724  //!
725  //! Tables of Bloch coefficients used to calculate the factors phi, lambda
726  //! and A_x should be passed as an argument to this function
727  ////////////////////////////////////////////////////////////////////////////////
728  void InteractingOflModel::ChangeToWannierBasis(
729  NonInteractingOflModel* blochTable, //!< Lookup table of Bloch wave functions
730  const double matrixElementTol, //!< Tolerance for setting matrix elements to precisely zero
731  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
732  {
733  if(0 == mpi.m_id) // FOR THE MASTER NODE
734  {
735  utilities::cout.MainOutput()<<"\n\tRECALCULATING COEFFICIENTS FOR WANNIER BASIS CASE"<<std::endl;
736  }
737  // Clear any existing hash tables, in preparation for population with new
738  // sets of coefficients
739  m_quadraticHashTables.Clear();
740  m_quarticHashTables.Clear();
741  m_quadraticHashTables.Initialize(m_params.m_dimX*m_params.m_dimY);
742  m_quarticHashTables.Initialize(m_params.m_dimX*m_params.m_dimY);
743  // Generate tables of angular momentum conserving values for the Wannier basis
744  this->GenerateWannier2KTable(m_quadraticHashTables.GetKTable(), mpi);
745  this->GenerateWannier4KTable(m_quarticHashTables.GetKTable(), mpi);
746  m_params.m_tableFormat = _HASH_;
747 
748  ////// Recalculate quadratic terms ////////////////////////////////////
749  // The new set of quadratic terms will be of the general form
750  // E_{x1,x2,ky}. It will be diagonal in ky still, but non-diagonal in x.
751  // sum_{kx,ky} E_{kx,ky} c^+_{kx,ky} c_{kx,ky}
752  // = sum_{x1,x2,ky} [sum_{kx} E_{kx,ky} |1/coefficient(kx,ky)|^2 1/dimX
753  // e^{2*I*PI*kx/dimX [x1-x2]}] c^+_{x1,ky} c_{x2,ky}
754  // The || of the coefficients is just 1 (since these coefficients are
755  // just phase factors. So we have:
756  //
757  // E_{x1,x2,ky} = e^{2*I*PI*offsetX/dimX [x1-x2]} sum_{kx} E_{kx,ky} 1/dimX e^{2*I*PI*kx/dimX [x1-x2]}
758  //
759  // The sums can be performed by a set of N_y discrete Fourier transforms
760  // The Fourier coefficients depend only on the difference x2-x1
761  // so we need only calculate the transform for all possible values
762  // of this difference (by Hermiticity we only need to consider x2>=x1).
763  {
764  if(0 == mpi.m_id) // FOR THE MASTER NODE
765  {
766  utilities::cout.SecondaryOutput()<<"\n\t- RECALCULATING QUADRATIC TERMS\n"<<std::endl;
767  }
768  utilities::LoadBar progress;
769  if(0 == mpi.m_id) // FOR THE MASTER NODE
770  {
771  progress.Initialize(m_params.m_dimX*m_params.m_dimY);
772  }
773  // Declare some variables to store input/outputs of Fourier transforms
774  dcmplx* fourierCoefficients = new dcmplx[m_params.m_dimX*m_params.m_dimY];
775  dcmplx* outputCoefficients = new dcmplx[m_params.m_dimX*m_params.m_dimY];
776  for(kState_t ky=0; ky<m_params.m_dimY; ++ky)
777  {
778  for(kState_t kx=0; kx<m_params.m_dimX; ++kx)
779  {
780  fourierCoefficients[ky*m_params.m_dimX+kx] = m_quadraticTables.GetVTable()->at(kx*m_params.m_dimY+ky)/((double)m_params.m_dimX);
781  if(0 == mpi.m_id) // FOR THE MASTER NODE
782  {
783  progress.Display(ky*m_params.m_dimX+kx+1);
784  }
785  }
786  }
787  if(0 == mpi.m_id) // FOR THE MASTER NODE
788  {
789  utilities::cout.AdditionalInfo()<<std::endl;
790  }
791  utilities::DiscreteFourierTransform1D(m_params.m_dimX, m_params.m_dimY, fourierCoefficients,
792  outputCoefficients, FFTW_BACKWARD);
793  // The output from the discrete transform is a set of coefficients like E_{ky,x2-x1}
794  // Next, we use that output to populate the look-up hash tables E_{x1,x2,ky}
795  for(kState_t x1=0; x1<m_params.m_dimX; ++x1)
796  {
797  for(kState_t x2=0; x2<m_params.m_dimX; ++x2)
798  {
799  for(kState_t ky=0; ky<m_params.m_dimY; ++ky)
800  {
801  m_quadraticHashTables.GetVTable()->Insert(utilities::Key(x1*m_params.m_dimY+ky, x2*m_params.m_dimY+ky)) = exp(2.0*I*PI*fabs(x2-x1)*m_params.m_offsetX/(double)m_params.m_dimX)*outputCoefficients[ky*m_params.m_dimX+abs(x2-x1)];
802  }
803  }
804  }
805  delete[] fourierCoefficients;
806  delete[] outputCoefficients;
807  }
808  ////// Recalculate quartic terms //////////////////////////////////////
809  // The new set of quadratic terms will be of the general form
810  //
811  // V_{x1,ky1,x2,ky2,x3,ky3,x4,ky4}.
812  //
813  // The list will contain a larger number of terms in the new basis
814  // because there is no conservation of the x values (so we need to
815  // do the full sum over x1,x2,x3,x4). One of the ky values remains
816  // fixed by the other three, however, due to momentum conservation
817  {
818  if(0 == mpi.m_id) // FOR THE MASTER NODE
819  {
820  utilities::cout.SecondaryOutput()<<"\n\t- RECALCULATING QUARTIC TERMS\n"<<std::endl;
821  }
822  ////// Calculate and store the change of basis matrix
823  NonInteractingOflModelGrid modelGrid;
824  modelGrid.GenerateWannierCoefficients(m_params.m_dimX, m_params.m_dimY, m_params.m_offsetX, blochTable,
825  &m_basisChangeMatrix[0], mpi);
826  iSize_t dimBasisChange = m_params.m_dimX*m_params.m_dimY;
827  // Reserve a memory block with which to retrieve the momentum conserving list of k-values
828  kState_t kRetrieveBuffer[m_quarticHashTables.GetMaxKCount()];
829  // The change of basis calculation is parallelized so that the
830  // outermost sums (over x4 and k4y) are distributed over the available
831  // nodes
832  // The hash tables are combined at the end of the calculation
833  mpi.DivideTasks(mpi.m_id, m_params.m_dimX*m_params.m_dimY, mpi.m_nbrProcs,
834  &mpi.m_firstTask, &mpi.m_lastTask, false);
835  iSize_t counter = 0;
836  utilities::LoadBar progress;
837  if(0 == mpi.m_id) // FOR THE MASTER NODE
838  {
839  progress.Initialize(mpi.m_lastTask+1);
840  }
841  for(kState_t x4=0; x4<m_params.m_dimX; ++x4)
842  {
843  for(kState_t k4y=0; k4y<m_params.m_dimY; ++k4y, ++counter)
844  {
845  kState_t k4 = x4*m_params.m_dimY+k4y;
846  if(mpi.m_firstTask <= (int)counter && mpi.m_lastTask >= (int)counter)
847  {
848  if(0 == mpi.m_id) // FOR THE MASTER NODE
849  {
850  progress.Display(counter+1);
851  }
852  for(kState_t x3=0; x3<m_params.m_dimX; ++x3)
853  {
854  for(kState_t k3y=0; k3y<m_params.m_dimY; ++k3y)
855  {
856  kState_t k3 = x3*m_params.m_dimY+k3y;
857  for(kState_t x2=0; x2<m_params.m_dimX; ++x2)
858  {
859  for(kState_t k2y=0; k2y<m_params.m_dimY; ++k2y)
860  {
861  kState_t k2 = x2*m_params.m_dimY+k2y;
862  // Get the corresponding list of k1 values
863  // that satisfy y momentum conservation
864  iSize_t nbrK1;
865  m_quarticHashTables.GetK1(kRetrieveBuffer,nbrK1,k2,k3,k4);
866  for(iSize_t j=0; j<nbrK1; ++j)
867  {
868  kState_t k1 = kRetrieveBuffer[j];
869  kState_t k1y = utilities::Modulo<int>(k1, m_params.m_dimY);
870  kState_t x1 = k1/m_params.m_dimY;
871  dcmplx summation = 0.0;
872  for(kState_t k4x=0; k4x<m_params.m_dimX; ++k4x)
873  {
874  for(kState_t k3x=0; k3x<m_params.m_dimX; ++k3x)
875  {
876  for(kState_t k2x=0; k2x<m_params.m_dimX; ++k2x)
877  {
878  kState_t q1;
879  kState_t q2 = k2x*m_params.m_dimY+k2y;
880  kState_t q3 = k3x*m_params.m_dimY+k3y;
881  kState_t q4 = k4x*m_params.m_dimY+k4y;
882  // Set the value of the q1 state using
883  // kx,ky momentum conservation
884  // (note that ky momentum conservation
885  // is already satisfied by k1y)
886  iSize_t nbrQ1;
887  m_quarticTables.GetK1(&q1, nbrQ1, q2, q3, q4);
888  // Note that this value is unique
889  unsigned int uIndex1 = (x1*m_params.m_dimY+k1y)*dimBasisChange+q1;
890  unsigned int uIndex2 = (x2*m_params.m_dimY+k2y)*dimBasisChange+q2;
891  unsigned int uIndex3 = (x3*m_params.m_dimY+k3y)*dimBasisChange+q3;
892  unsigned int uIndex4 = (x4*m_params.m_dimY+k4y)*dimBasisChange+q4;
893  summation += m_quarticTables.GetVkkkk(q1, q2, q3, q4)*std::conj(m_basisChangeMatrix[uIndex1]*m_basisChangeMatrix[uIndex2])*m_basisChangeMatrix[uIndex3]*m_basisChangeMatrix[uIndex4];
894  }
895  }
896  }
897  if(abs(summation)>matrixElementTol)
898  {
899  m_quarticHashTables.GetVTable()->Insert(utilities::Key(k1, k2, k3, k4)) = summation;
900  }
901  }
902  }
903  }
904  }
905  }
906  }
907  }
908  }
909  if(0 == mpi.m_id) // FOR THE MASTER NODE
910  {
911  utilities::cout.AdditionalInfo()<<std::endl;
912  }
913  }
914  // Combine Hash tables populated on each node
915  m_quarticHashTables.GetVTable()->MpiGather(0, mpi);
916  m_quadraticHashTables.GetVTable()->MpiGather(0, mpi);
917  m_quarticHashTables.GetVTable()->MpiSynchronize(0, mpi);
918  m_quadraticHashTables.GetVTable()->MpiSynchronize(0, mpi);
919  // Clear existing non-hash tables
920  m_quarticTables.Clear();
921  m_quadraticTables.Clear();
922  MPI_Barrier(mpi.m_comm);
923  }

1420  {
1421  if(!m_params.m_hamiltonianDiagonalized)
