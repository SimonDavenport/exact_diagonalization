34  //! Get the highest number of k1 values that will be returned
35  //! for a given k2. This allows us to pre-allocate a memory buffer
36  //! to store the return values statically
37  //!
38  iSize_t QuadraticTermHashTables::GetMaxKCount() const
39  {
40  iSize_t maxCount = 1;
41  for(kState_t k2=0; k2<m_kMax; ++k2)

43  maxCount = std::max(maxCount, (iSize_t)m_kTable.Count(k2));

46  }
