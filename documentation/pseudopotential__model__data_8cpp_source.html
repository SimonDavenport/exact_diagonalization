179  const int nodeId, //!< Node id to sync with
180  const utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
181  {
182  mpi.Sync(&m_nbrOrbitals, 1, nodeId);
183  mpi.Sync(&m_nbrParticles, 1, nodeId);
184  mpi.Sync(&m_maxLz, 1, nodeId);
185  mpi.Sync(&m_maxLz2, 1, nodeId);
186  mpi.Sync(&m_totalLz, 1, nodeId);
187  mpi.Sync(&m_nbrLevels, 1, nodeId);
188  mpi.Sync(&m_twoBodyPseudopotentials, nodeId);
189  mpi.Sync(&m_twoBodyPseudopotentials2LL, nodeId);
190  mpi.Sync(m_inPath, nodeId);
191  mpi.Sync(m_outPath, nodeId);
192  mpi.Sync(m_outFileName, nodeId);
193  mpi.Sync(&m_nbrEigenvaluesToFind, 1, nodeId);
194  mpi.Sync(&m_blockDiagonalize, 1, nodeId);
195  int method = 0 ;
196  if(_FULL_ == m_method)
197  {
198  method = 0;
199  }
200  else if(_LANCZOS_ == m_method)
201  {
202  method = 1;
203  }
204  mpi.Sync(&method, 1, nodeId);
205  if(0 == method)
206  {
207  m_method = _FULL_;
208  }
209  else if(1 == method)
210  {
211  m_method = _LANCZOS_;
212  }
213  mpi.Sync(m_initialVectorFile, nodeId);
214  mpi.Sync(m_finalVectorFile, nodeId);
215  mpi.Sync(&m_termTablesBuilt, 1, nodeId);
216  mpi.Sync(&m_fockBasisBuilt, 1, nodeId);
217  mpi.Sync(&m_hamiltonianBuilt, 1, nodeId);
218  mpi.Sync(&m_hamiltonianDiagonalized, 1, nodeId);
219  // Barrier waits for all nodes to catch up to this point
220  MPI_Barrier(mpi.m_comm);
221  }

419  //!
420  //! Set the minimum value of Lz such that the orbitals are labelled by
421  //! -m_maxLz to m_maxLz in steps of 2
422  //!
423  void PseudopotentialModelData::SetMaxLz(
424  const iSize_t nbrOrbitals, //!< Number of orbitals
425  const iSize_t nbrLevels) //!< Number of Landau levels
426  {
427  if(nbrLevels==1)
428  {
429  // Label each level in the same way and treat independently
430  m_maxLz = m_nbrOrbitals-1;
431  m_maxLz2 = m_nbrOrbitals-1;
432  }
433  else if(nbrLevels==2)
434  {
435  // Line up the angular momentum labels of the two levels
436  iSize_t nbrOrbitalsLLL = (m_nbrOrbitals-2)/2;
437  m_maxLz = nbrOrbitalsLLL-1;
438  m_maxLz2 = nbrOrbitalsLLL+1;
439  }
440  }
