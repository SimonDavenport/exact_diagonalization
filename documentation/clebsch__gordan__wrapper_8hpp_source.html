84  ////////////////////////////////////////////////////////////////////////////////
85  //! \brief Generate a set of tables of Clebsch-Gordan coefficients
86  //! for multiplet labels j1 and j2 only and add to the current hash
87  //! table of values.
88  //!
89  //! NOTE: the maximum quanutm number value is 128 (represented by input j1=256).
90  //! Any larger number will result in a truncation when converted to a j_type
91  //! variable
92  ////////////////////////////////////////////////////////////////////////////////
93  void AddMultiplet(
94  const j_type j1, //!< j1 between 0 and 256 represeting 2*j1
95  const j_type j2) //!< j1 between 0 and 256 represeting 2*j2
96  {
97  // Call the "clebsch" code to calculate all relevant SU(2) CG coefficients
98  // then reassign the CG coefficients to a more convenient hash map
99  // indexes by conventional SU(2) weights and roots.
100  clebsch::Weight S(2);
101  S(1) = j1;
102  S(2) = 0;
103  clebsch::Weight Sprime(2);
104  Sprime(1) = j2;
105  Sprime(2) = 0;
106  clebsch::Decomposition decomp(S, Sprime);
107  for(j_type d = 0; d < decomp.Size(); ++d)
108  {
109  const clebsch::Coefficients C(decomp(d),S, Sprime);
110  j_type dimS = S.Dimension();
111  j_type dimSprime = Sprime.Dimension();
112  j_type dimSdoubleprime = decomp(d).Dimension();
113  for(j_type m = 0; m < C.multiplicity; ++m)
114  {
115  for(j_type m3 = 0; m3 < dimSdoubleprime; ++m3)
116  {
117  for(j_type m1 = 0; m1 < dimS;++m1)
118  {
119  for(j_type m2 = 0; m2 < dimSprime; ++m2)
120  {
121  double x = double(C(m1, m2, m, m3));
122  if(fabs(x) > clebsch::EPS)
123  {
124  j_type j3 = decomp(d)(1);
125  m_map.Insert(utilities::Key(j1, m1, j2, m2, j3, m3)) = x;
126  }
127  }
128  }
129  }
130  }
131  }
132  return;
133  }

172  return m_map.Value(utilities::Key(j1,unsignedM1,j2,unsignedM2,j3,unsignedM3));
173  }
