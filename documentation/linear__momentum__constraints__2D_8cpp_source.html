32  //! \brief A function to test whether a given Fock state is in a given quantum
33  //! number sector.
34  //!
35  //! This function defines the representation of the Fock states in terms of
36  //! linear momentum quantum numbers kx,ky.
37  ////////////////////////////////////////////////////////////////////////////////
38  bool TestLinearMomentumSector2(
39  const fock_t state, //!< Fock state to be tested
40  const kState_t kxTot, //!< Given linear momentum sector
41  //!< To test against
42  const kState_t kyTot, //!< Given linear momentum sector
43  //!< To test against
44  const kState_t dimX, //!< X dimension of k-space grid
45  const kState_t dimY) //!< Y dimension of k-space grid
46  {
47  // The occupation basis is of 64-bit integer format e.g. 000100101...
48  // Let's say that the ith bit from the right corresponds to the
49  // kx*ny+ky th state, where ny is the number of ky states.
50  // e.g. the i=3 state (with ny=4 say) corresponds to kx=0,ky=3
51  // the i=5 state would in that case be kx=1,ky=0 etc.
52  // States in a given sector have the same total value of kx and
53  // ky (modulo nx or ny respectively)
54  // The number of sectors is equal to nx*ny. We shall label different
55  // sectors by a single quantum number kxTot*ny + kyTot
56  // Determine the kx_tot and ky_tot for the specified state
57  fock_t tempState = state;
58  int x = 0;
59  int y = 0;
60  while(0 != tempState)
61  {
62  // Isloate right-most bit and determine its position
63  fock_t rightMost = tempState & -tempState;
64  kState_t position = utilities::binary::Base2Log(rightMost);
65  // Convert to linear momentum quantum numbers
66  // and accumulate total linear momentum
67  x += floor((double)position/(dimY));
68  y += utilities::Modulo<int>(position, dimY);
69  // Mask off the current right-most bit and iterate
70  tempState = tempState ^ rightMost;
71  }
72  x = utilities::Modulo<int>(x, dimX);
73  y = utilities::Modulo<int>(y, dimY);
74  x = x<0 ? x + dimX : x;
75  y = y<0 ? y + dimY : y;
76  // Compare the specified sector with the given linear momentum
77  // sector
78  return (x == kxTot && y == kyTot);
79  }

123  //! \brief A function to add two k-states to give a result that is periodic
124  //! in the unit cell
125  //!
126  //! \return k1+k2 pariodic in 2D momentum space
127  ////////////////////////////////////////////////////////////////////////////////
128  kState_t StateAddition(
129  const kState_t k1, //!< First k state
130  const kState_t k2, //!< First k state
131  const kState_t dimX1, //!< X dimension of k-space grid

171  //! \brief A function to impose a number of R60 rotations on a given k state
172  //!
173  //! \return R60 operator applied r times to the state
174  ////////////////////////////////////////////////////////////////////////////////
175  kState_t R60(
176  const kState_t k, //!< Input k state
177  const int r, //!< Number of times to apply the rotation
178  const kState_t dimX, //!< X dimension of k-space grid
179  const kState_t dimY) //!< Y dimension of k-space grid
180  {
181  int kx = floor((double)k/(dimY));
182  int ky = utilities::Modulo<int>(k, dimY);
183  int k1x = 0;
184  int k1y = 0;
185  switch(utilities::Modulo<int>(r, 6))
186  {
187  case 0:
188  k1x = kx;
189  k1y = ky;
190  break;
191  case 1:
192  k1x = utilities::Modulo<int>(kx+ky, dimX);
193  k1y = utilities::Modulo<int>(-kx, dimY);
194  break;
195  case 2:
196  k1x = utilities::Modulo<int>(ky, dimX);
197  k1y = utilities::Modulo<int>(-kx-ky, dimY);
198  break;
199  case 3:
