71  //!

96  GetOption(optionList, sqlId, "sql-id", _AT_, mpi);
97  fileName << inPath << sqlName;
98  this->ReadFromSql(sqlTableName, fileName.str(), sqlId, mpi);
99  if(mpi.m_exitFlag)
100  {
101  return;
102  }
103  }
104  else
105  {
106  // Default to look for model data in the specified text file

108  GetOption(optionList, paramsFile, "params-file", _AT_, mpi);
109  fileName << inPath << paramsFile;
110  this->ReadFromFile(fileName.str(), mpi);

112  {
113  return;
114  }
115  }
116  // Print out summary of model parameters:
117  utilities::cout.MainOutput()<<"\n\tNON-INTERACTING MODEL PARAMETERS:\n"<<std::endl;

128  //!
129  //! Read model parameters for a specified file.
130  //!
131  void NonInteractingOflModelData::ReadFromFile(
132  std::string fileName, //!< Name of file where parameters are contained
133  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
134  {

177  //!
178  //! Function to read parameters from a sql database
179  //!
180  void NonInteractingOflModelData::ReadFromSql(
181  const std::string tableName, //!< Name of SQL table where parameters are contained
182  const std::string fileName, //!< Name of file where parameters are contained
183  const iSize_t sqlId, //!< Sql identifier of the parameter set
184  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
185  {
186  utilities::Sqlite sql(fileName, sql::_READ_EXISTING_);
187  mpi.m_exitFlag = !sql.IsOpen();
188  if(mpi.m_exitFlag)
189  {
190  return;
191  }
192  utilities::SqliteRow sqlRow = sql.RetrieveIdFromTable(tableName, sqlId);
193  if(sqlRow.GetLength() == 0)
194  {
195  mpi.m_exitFlag = true;
196  return;
197  }
198  else
199  {
200  utilities::cout.MainOutput()<<"\n\tReading model parameters from SQLite database "<<fileName<<std::endl;
201  }
202  mpi.m_exitFlag = sqlRow.GetValue("Theta", m_theta);
203  if(mpi.m_exitFlag)
204  {
205  return;
206  }
207  mpi.m_exitFlag = sqlRow.GetValue("V0", m_V0);
208  if(mpi.m_exitFlag)
209  {
210  return;
211  }
212  mpi.m_exitFlag = sqlRow.GetValue("Epsilon", m_epsilon);
213  if(mpi.m_exitFlag)
214  {
215  return;
216  }
217  mpi.m_exitFlag = sqlRow.GetValue("Kappa", m_kappa);
218  if(mpi.m_exitFlag)
219  {
220  return;
221  }
222  mpi.m_exitFlag = sqlRow.GetValue("Mass", m_mass);
