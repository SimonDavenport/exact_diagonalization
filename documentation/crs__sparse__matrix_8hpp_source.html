57  //////////////////////////////////////////////////////////////////////////////////
58  //! \brief Comparison operator to test if two CrsSparseMatrix data structures
59  //! are identical up to a given tolerence. Only classes with the same template
60  //! arguments can be compared.
61  //!
62  //! Note that the comparison algorithm requires an in-place sort of the
63  //! underlying data and column arrays in the CrsSparseMatrix class. This
64  //! does not change the data represented.
65  //////////////////////////////////////////////////////////////////////////////////
66  template<typename T>
67  bool SameTest(
68  CrsSparseMatrix<T>& lhs, //!< lhs matrix to compare
69  CrsSparseMatrix<T>& rhs, //!< rhs matrix to compare
70  const double tol) //!< comparison tolerence
71  {
72  if(
73  (lhs.GetNbrNonZeros() != rhs.GetNbrNonZeros()) ||
74  (lhs.m_rowStarts != rhs.m_rowStarts)
75  )
76  {
77  return false;
78  }
79  // In CRS format the row data is always stored in order
80  for(int i=0; i<lhs.GetNbrRows(); ++i)
81  {
82  // For each row, we need to ensure that the column indices are
83  // in the same order (since this is not necessarily guaranteed)
84  // So run a quick sort to rearrange the columns and their
85  // corresponding data entries
86  int startCol = lhs.m_rowStarts[i];
87  int endCol = lhs.m_rowStarts[i+1];
88  utilities::QuickSort<int, T, _ASCENDING_ORDER_>(&lhs.m_cols[startCol],
89  &lhs.m_data[startCol], endCol-startCol);
90  utilities::QuickSort<int, T ,_ASCENDING_ORDER_>(&rhs.m_cols[startCol],
91  &rhs.m_data[startCol], endCol-startCol);
92  }
93  // Now test for equality of the underlying data structures
94  if(lhs.m_cols != rhs.m_cols)
95  {
96  return false;
97  }
98  if(utilities::is_same<dcmplx, T>::value)
99  {
100  for(int i=0; i<lhs.GetNbrNonZeros(); ++i)
101  {
102  if(abs(lhs.m_data[i]-rhs.m_data[i])>tol) return false;
103  }
104  }
105  else if(utilities::is_same<double,T>::value)
106  {
107  for(int i=0; i<lhs.GetNbrNonZeros(); ++i)
108  {
109  if(fabs(lhs.m_data[i]-rhs.m_data[i])>tol) return false;
110  }
111  }
112  return true;
113  }

182  //////////////////////////////////////////////////////////////////////////////////
183  //! \brief Initialize from MappedSparseMatrix. The data values must be the same
184  //! type T
185  //!
186  //! NOTE 1: This initialization function will dealocate the original
187  //! MappedSparseMatrix data structure in the process.
188  //!
189  //! NOTE 2: the data in the hash table implementation is assumed to be unordered
190  //! in general (it would be more efficient if it were ordered). This function
191  //! performs a sorting operation on the map elements before converting to CRS format
192  //////////////////////////////////////////////////////////////////////////////////
193  void Initialize(
194  MappedSparseMatrix<T>* other) //!< Mapped matrix data structure
195  {
196  // First call the std::sort algorithm on the unordered list
197  // of non-zero matrix elements
198  // We need to extract the map data as an array of pairs
199  // Allocate memory to store the ENTIRE map
200  int nnz = other->m_map.size();
201  pair<uint64_t, T>* sortingBuffer = new pair<uint64_t, T>[nnz];
202  pair<uint64_t, T>* p_buffer = sortingBuffer;
203  // Set the pair values
204  for(auto& x: other->m_map)
205  {
206  p_buffer->first = x.first;
207  p_buffer->second = x.second;
208  ++p_buffer;
209  }
210  // Deallocate memory associated with the original map
211  other->m_map.clear();
212  #if _SPEED_OPTIMIZED_MAP_ || _MEMORY_OPTIMIZED_MAP_
213  other->m_map.resize(0);
214  #endif
215  // Apply a std::sort algorithm on the array of pairs
216  // (sorting in order of the map key values)
217  std::sort(sortingBuffer, sortingBuffer+nnz);
218  // Now convert the sorted arrays into CRS format
219  m_cols.reserve(nnz);
220  m_data.reserve(nnz);
221  // Generate an array of row start "pointers" initialized to zero
222  m_rowStarts.resize(other->m_nbrRows+1);
223  m_rowStarts.back() = nnz;
224  int rowCounter = 0;
225  int currRow = 0;
226  p_buffer = sortingBuffer;
227  for(int i=0;i<nnz; ++i, ++p_buffer)
228  {
229  uint32_t row;
230  uint32_t col;
231  utilities::Unpack2x32(p_buffer->first, row, col);
232  m_cols.push_back(col);
233  m_data.push_back(p_buffer->second);
234  if(row >= currRow)
235  {
236  // Increment to the actual row value
237  while(currRow<=row)
238  {
239  m_rowStarts[currRow] = rowCounter;
240  ++currRow;
241  }
242  }
243  ++rowCounter;
244  }
245  // Set the remaining row starts to the final counter value
246  while(currRow<other->m_nbrRows)
247  {
248  m_rowStarts[currRow] = rowCounter;
249  ++currRow;
250  }
251  // Deallocate the sorting buffer
252  delete[] sortingBuffer;
253  return;
254  }

342  //////////////////////////////////////////////////////////////////////////////////
343  //! \breif Accessor function to read matrix element value
344  //!
345  //! Use as e.g.: element = object.value(i,j); or std::cout<<object.value(i,j); etc.
346  //!
347  //! \return The value of the matrix element i,j (or 0.0 if the
348  //! element could not be found)
349  //////////////////////////////////////////////////////////////////////////////////
350  const T Value(
351  const int i, //!< row index
352  const int j) //!< column index
353  const
354  {
355  // Since the array is unordered in general, we cannot use
356  // a fast binary search to find the matrix element
357  // (use the mapped format for fast indexing)
358  if(i<0 || i >= m_rowStarts.size()-1)
359  {
360  std::cerr<<"\n\tERROR in CrsSparseMatrix with accessing element ["<<i<<","<<j<<"]"<<std::endl;
361  std::cerr<<"\tINDEX OUT OF BOUNDS"<<std::endl;
362  return 0.0;
363  }
364  else
365  {
366  for(int t = m_rowStarts[i]; t<m_rowStarts[i+1]; ++t)
367  {
368  if(m_cols[t] == j) return m_data[t];
369  }
370  }
371  // If it was not found, then the element is zero
372  return 0.0;
373  }

376  //! \brief Accessor function to insert or update matrix element value
377  //!
378  //! Use as: object.insert(i,j) = element; OR object.insert(i,j) += element; etc.

382  //! position and the address of the new element is returned.
383  //////////////////////////////////////////////////////////////////////////////////
384  T& Insert(
385  const int i, //!< row index
386  const int j) //!< column index
387  {
388  // Since the array is unordered in general, we cannot use
389  // a fast binary search to find the matrix element
390  // (use the mapped format for fast indexing)
391  if(i<0 || i >= m_rowStarts.size()-1)

397  else

408  for(int r = i+1; r<m_rowStarts.size(); ++r)
409  {
410  ++m_rowStarts[r];
411  }
412  return m_data[m_rowStarts[i+1]-1];

416  //////////////////////////////////////////////////////////////////////////////////
417  //! \brief MPI gather function
418  //!
419  //! Combine instances of this class together onto a single instance on
420  //! the master node. Memory allocations on remaining nodes are then cleared.
421  //!
422  //! NOTE: the row indices on each node are assumed to be independent.
423  //! So e.g. rows 1-100 on node 0 + rows 1-100 on node 1 become rows
424  //! 1-200 on node 0.
425  //////////////////////////////////////////////////////////////////////////////////
426  void MpiGather(
427  const int gatherNode, //!< Node to gather data onto
428  const MpiWrapper& mpi) //!< Instance of the MPI wrapper class
429  {
430  // First we determine how many rows there will be in the combined matrix
431  // With the CRS Format in parallel, it is assumed that some number
432  // of matrix rows are stored on each node in order
433  int totalNbrRows = 0;
434  int nbrRows = m_rowStarts.size()-1;
435  MPI_Reduce(&nbrRows,&totalNbrRows, 1, mpi.GetType<int>(), MPI_SUM, gatherNode, mpi.m_comm);
436  // Second we need the total number of non-zero elements and the
437  // cumulative number stored on each node in order
438  int nbrNonZeros = m_data.size();
439  int totalNbrNonZero = 0;
440  int allNbrNonZero[mpi.m_nbrProcs];
441  int cumulativeNbrNonZero = 0;
442  MPI_Reduce(&nbrNonZeros, &totalNbrNonZero, 1, mpi.GetType<int>(), MPI_SUM, gatherNode, mpi.m_comm);
443  MPI_Allgather(&nbrNonZeros, 1, mpi.GetType<int>(), &allNbrNonZero, 1, mpi.GetType<int>(), mpi.m_comm);
444  for(int k=0; k<mpi.m_id; ++k)
445  {
446  cumulativeNbrNonZero += allNbrNonZero[k];
447  }
448  // Shift the row starts by the cumulative number of non-zeros
449  // on preceding nodes
450  for(int i=0; i<nbrRows; ++i)
451  {
452  m_rowStarts[i] += cumulativeNbrNonZero;
453  }
454  MPI_Barrier(mpi.m_comm);
455  // Next we combine the index and data arrays from each node
456  // It is assumed that each node contains a set of matrix rows
457  // in the correct order, and no row sorting takes place in the
458  // gathered matrix
459  if(gatherNode == mpi.m_id)
460  {
461  utilities::cout.DebuggingInfo()<<"\n\t- ON GATHER NODE: REALLOCATING (ESTIMATED) "<<((totalNbrRows*sizeof(int)+totalNbrNonZero*(sizeof(T)+sizeof(int)))/(1024.0*1024.0));
462  utilities::cout.DebuggingInfo()<<" MB TO STORE ENTIRE SPARSE MATRIX"<<std::endl;
463  m_rowStarts.resize(totalNbrRows+1);
464  m_cols.resize(totalNbrNonZero);
465  m_data.resize(totalNbrNonZero);
466  m_rowStarts.back() = totalNbrNonZero;
467  }
468  MPI_Status status;
469  mpi.Gather<int>(m_rowStarts.data(), nbrRows, m_rowStarts.data(), totalNbrRows,
470  gatherNode, mpi.m_comm, status);
471  mpi.Gather<int>(m_cols.data(), nbrNonZeros, m_cols.data(), totalNbrNonZero,
472  gatherNode, mpi.m_comm, status);
473  mpi.Gather<T>(m_data.data(), nbrNonZeros, m_data.data(), totalNbrNonZero, gatherNode, mpi.m_comm, status);
474  MPI_Barrier(mpi.m_comm);
475  }

477  //////////////////////////////////////////////////////////////////////////////////
478  //! \brief A function to resize the row starts array such that the stored
479  //! matrix now contains an additional number of empty rows after the
480  //! populated part - assuming that the matrix is stored
481  //! in parallel (i.e. with some number of rows stored on each node).
482  //! This turns it into a square matrix, which is required for the matrix
483  //! vector multiplication routine.
484  //!
485  //! The picture to have in mind is as follows. We solve Y = A X, where
486  //! the memory distribution of the vectors/matrix is as:
487  //!
488  //! +------------+
489  //! Y0 | A0 | X0 } dataDistribution[0]
490  //! | +--------+
491  //! Y1 | | A1 | X1 } dataDistribution[1]
492  //! | | +----+
493  //! Y2 | | | A2 | X2 } dataDistribution [2]
494  //!
495  //! Where the first dataDistribution[0] rows of the matrix are stored on
496  //! node 0, and so on. Note that the Hermitian structure of the matrix
497  //! means that the first n rows also give the first n columns.
498  //!
499  //! The matrix A0 can be thought of as a NxN matrix (N the total dimension)
500  //! with zeros where A1, A2 are labelled. A1 is a (N-dataDistribution[0])x
501  //! (N-dataDistribution[0]) matrix with zeros where A2 is labelled, and so on.
502  //! Appropriate restructuring is done by PrepareForParallelMultiplication.
503  //////////////////////////////////////////////////////////////////////////////////
504  void PrepareForParallelMultiplication(
505  std::vector<int>* dataDistribution, //!< A vector to store the number
506  //! of rows on each node
507  std::vector<MPI_Comm>* commGroups, //!< List of processor groups
508  //! required for minimal
509  //! vector MPI communication
510  std::vector<T>* inputVectorBuffer, //!< Buffer for MPI distribution of X vector
511  std::vector<T>* outputVectorBuffer, //!< Buffer for MPI accumulation of Y vector
512  const MpiWrapper& mpi) //!< Instance of the MPI wrapper class
513  {
514  // First we determine how many rows are stored on each node, the
515  // the cumulative number stored and the total number stored. This
516  // information is required on all nodes.
517  if(!m_preparedForParallelMultiply)
518  {
519  int cumulativeRows = 0;
520  int totalRows;
521  int nbrRows = m_rowStarts.size()-1;
522  dataDistribution->resize(mpi.m_nbrProcs);
523  MPI_Allreduce(&nbrRows, &totalRows, 1, mpi.GetType<int>(), MPI_SUM, mpi.m_comm);
524  MPI_Allgather(&nbrRows, 1, mpi.GetType<int>(), &(*dataDistribution)[0], 1,
525  mpi.GetType<int>(), mpi.m_comm);
526  for(int k=0; k<mpi.m_id; ++k)
527  {
528  cumulativeRows += (*dataDistribution)[k];
529  }
530  int newSize = totalRows-cumulativeRows;
531  // insert an array for the empty rows below the populated part
532  int nnz = m_rowStarts.back();
533  m_rowStarts.resize(newSize+1, nnz);
534  // Update the m_cols array to shrink the matrix size stored on each node
535  for(int i=0; i<nnz; ++i)
536  {
537  m_cols[i] -= cumulativeRows;
538  }
539  // Define a list of processor groups that will allow
540  // minimal MPI communication during vector multiplication
541  for(int i = 0; i<mpi.m_nbrProcs; ++i)
542  {
543  // Broadcast the contents of the buffer to the preceding i nodes
544  // only, by splitting up the MPI communicator into 2 groups -
545  // one for the preceding nodes, one for the rest
546  MPI_Comm minimalComm;
547  int whichGroup = (i>=mpi.m_id ? 0 : 1);
548  MPI_Comm_split(mpi.m_comm, whichGroup,mpi.m_id, &minimalComm);
549  commGroups->push_back(minimalComm);
550  }
551  inputVectorBuffer->resize(newSize);
552  outputVectorBuffer->resize(newSize);
553  m_preparedForParallelMultiply = true;
554  }
555  }

753  ////////////////////////////////////////////////////////////////////////////////
754  //! \brief A function to remove any very small elements from the matrix
755  //! to save on storage space (slow)
756  //!
757  //! This function uses c++11 methods - compile with the --c++11 option
758  ////////////////////////////////////////////////////////////////////////////////
759  void Trim(
760  const double sparseZeroTol) //!< Tolerance to assume matrix elements are zero
761  {
762  // Iterate over matrix elements and test for zero values
763  // First update the column index list and list of row starts
764  // Declare variables to pass to a c++11 "lambda function"
765  std::vector<int> newRowStarts(m_rowStarts);
766  auto data_it = m_data.begin();
767  auto row_it = m_rowStarts.begin();
768  auto newRow_it = newRowStarts.begin();
769  int nbrRows = m_rowStarts.size()-1;
770  int counter = 0;
771  auto it = std::remove_if(m_cols.begin(), m_cols.end(),
772  // Declare lambda function [variables](input){function body}
773  [sparseZeroTol, &data_it, &counter, &row_it, &newRow_it, nbrRows](int x) -> bool
774  {
775  bool tmp = false;
776  if(utilities::is_same<T,dcmplx>::value)
777  {
778  tmp = abs(*data_it) < sparseZeroTol;
779  }
780  else if(utilities::is_same<T,double>::value)
781  {
782  tmp = fabs(*data_it) < sparseZeroTol;
783  }
784  if(tmp)
785  {
786  for(int r=0; r<nbrRows-1; ++r)
787  {
788  if(counter>=row_it[r] && counter<row_it[r+1])
789  {
790  for(int k=r+1; k<nbrRows; ++k)
791  {
792  newRow_it[k]--;
793  }
794  break;
795  }
796  }
797  }
798  ++data_it;
799  ++counter;
800  return tmp;
801  }
802  ); // Complete the remove_if call
803  // Update the list of row start positions
804  m_rowStarts = newRowStarts;
805  // Clear the temporary list
806  newRowStarts.clear();
807  m_cols.erase(it, m_cols.end());
808  // Also update the data values
809  auto it1 = std::remove_if(m_data.begin(), m_data.end(),
810  [sparseZeroTol](dcmplx x){return abs(x) < sparseZeroTol;});
811  m_data.erase(it1, m_data.end());
812  // Finally, shrink the containers to exactly fit the remaining data
813  m_data.shrink_to_fit();
814  m_cols.shrink_to_fit();
815  }
