123  //! E_{k1,k2} c_k1^+ c_k2
124  //!
125  //! The function argument must be a class containing the following three methods:
126  //!
127  //! - GetMaxKCount()
128  //!
129  //! Return the maximum number of k1 values that will be returned given any k2

131  //!
132  //! - GetK1(kRetrieveBuffer,nbrK1,k2)
133  //!
134  //! To return a list of k1 values related to a given k2 by some conservation
135  //! law (where each k represents the pair kx,ky on a 2D grid).
136  //!
137  //! - GetEkk(k1,k2)
138  //!
139  //! To return a single Ekk coefficient for the specified k indices
140  //!

145  L* termTables, //!< Class container for terms tables
146  fock_t minRow, //!< Minimum row index to calculate for
147  fock_t maxRow, //!< Maximum row index to calculate for
148  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class

150  if(this->m_data.m_matrixAllocated)
151  {
152  utilities::Timer timer;
153  if(0 == mpi.m_id) // FOR THE MASTER NODE
154  {
155  utilities::cout.AdditionalInfo()<<"\n\t- ADDING C^+ C TERMS...\n"<<std::endl;
156  timer.Start(); // Time the operation
157  }
158  // For parallel processes, divide up tasks amongst available nodes
159  mpi.DivideTasks(mpi.m_id, this->m_data.m_fockSpaceDim,
160  mpi.m_nbrProcs, &mpi.m_firstTask, &mpi.m_lastTask, false);
161  fock_t lowerRowLimit = std::max((fock_t)mpi.m_firstTask, mpi.m_firstTask+minRow);
162  fock_t upperRowLimit = std::min((fock_t)mpi.m_lastTask, mpi.m_firstTask+maxRow);
163  // Reserve a memory block with which to retrieve the momentum conserving list of k-values
164  kState_t kRetrieveBuffer[termTables->GetMaxKCount()];
165  // Determine the starting Fock state on each node
166  std::vector<fock_t>::const_iterator it_state;
167  fock_t currState = this->m_fockBasis.GetFirstState(lowerRowLimit, this->m_data.m_nbrParticles, it_state);
168  utilities::LoadBar progress;
169  if(0 == mpi.m_id) // FOR THE MASTER NODE
170  {
171  progress.Initialize(this->m_data.m_nodeDim);
172  }
173  for(int i=lowerRowLimit; i<=upperRowLimit; ++i)
174  {
175  fock_t inState = currState;
176  // First, we can restrict the sum to k2 values that do no annihilate
177  // the inState
178  fock_t tempK2 = inState;
179  while(tempK2!=0) // N iterations for N particles
180  {
181  fock_t k2Occupied = tempK2 & -tempK2;
182  kState_t k2 = utilities::binary::HammingWeight64(k2Occupied-1);
183  iSize_t nbrK1;
184  termTables->GetK1(kRetrieveBuffer, nbrK1, k2);
185  // Loop over the possible k1 values
186  for(iSize_t j=0; j<nbrK1; ++j)
187  {
188  kState_t k1 = kRetrieveBuffer[j];
189  T matrixElement = 0.0;
190  fock_t outIndex = i;
191  if(k1==k2) // Case of diagonal elements
192  {
193  matrixElement = termTables->GetEkk(k1, k2);
194  }
195  else
196  {
197  // Zero if k1 occurs in the inState
198  fock_t k1Occupied = (fock_t)1 << k1;
199  if(k1Occupied & ~inState)
200  {
201  // Act on the current in state with CdC
202  int sign;
203  fock_t outState = utilities::Evaluate_CdC(sign, k1Occupied, k2Occupied, inState,
204  this->m_data.m_highestOrbital);
205  if(0 != sign && outState>=inState)
206  {
207  // If non zero then calculate the matrix element
208  matrixElement = termTables->GetEkk(k1, k2)*(double)sign;
209  outIndex = this->m_fockBasis.FindFockStateIndex(mpi.m_firstTask, outState);
210  }
211  }
212  }
213  // Don't bother to add the matrix element if it is zero
214  if(outIndex < this->m_data.m_fockSpaceDim)
215  {
216  if((utilities::is_same<dcmplx,T>::value && abs(matrixElement)>_SPARSE_ZERO_TOL_) || (utilities::is_same<double, T>::value && fabs(matrixElement)>_SPARSE_ZERO_TOL_))
217  {
218  if(utilities::_DENSE_ == this->m_data.m_storageMethod)
219  {
220  this->m_fullMatrix[(i-mpi.m_firstTask)*this->m_data.m_fockSpaceDim+outIndex] += matrixElement;
221  }
222  else if(utilities::_SPARSE_MAPPED_ == this->m_data.m_storageMethod)
223  {
224  this->m_sparseMap.Insert((i-mpi.m_firstTask),outIndex) += matrixElement;
225  }
226  else if(utilities::_SPARSE_CRS_ == this->m_data.m_storageMethod)
227  {
228  this->m_crsSparseMatrix.Insert((i-mpi.m_firstTask),outIndex) += matrixElement;
229  }
230  }
231  }
232  }
233  tempK2 = tempK2 ^ k2Occupied; // Remove right-most bit and proceed to the next iteration
234  }
235  currState = this->m_fockBasis.GetNextState(currState, it_state);
236  if(0 == mpi.m_id) // FOR THE MASTER NODE
237  {
238  progress.Display(i+1);
239  }
240  }
241  if(0 == mpi.m_id) // FOR THE MASTER NODE

260  //! \brief A template function to add all valid quartic terms of the form
261  //! V_{k1,k2,k3,k4} c_k1^+ c_k2^+ c_k3 c_k4
262  //!
263  //! The function argument must be a class containing the following three methods:
264  //!
265  //! - GetMaxKCount()
266  //!
267  //! Return the maximum number of k1 values that will be returned given any
268  //! k2,k3,k4 due to any kind of conservation law
269  //!
270  //! - GetK1(kRetrieveBuffer,nbrK1,k2,k3,k4)
271  //!
272  //! To return a list of k1 values related to a given k2,k3,k4 by some
273  //! conservation law (where each k represents the pair kx,ky on a 2D grid).
274  //!
275  //! - GetVkkkk(k1,k2,k3,k4)
276  //!
277  //! To return a single Vkkkk coefficient for the specified k indices
278  //!
279  //! This function MUST be called on all nodes if run in parallel
280  ////////////////////////////////////////////////////////////////////////////////
281  template <class L>
282  void Add_CdCdCC_Terms(
283  L* termTables, //!< Class container for terms tables
284  fock_t minRow, //!< Minimum row index to calculate for
285  fock_t maxRow, //!< Maximum row index to calculate for
286  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
287  {
288  if(this->m_data.m_matrixAllocated)
289  {
290  utilities::Timer timer;
291  if(0 == mpi.m_id) // FOR THE MASTER NODE
292  {
293  utilities::cout.AdditionalInfo()<<"\n\t- ADDING C^+ C^+ C C TERMS..."<<std::endl;
294 
295  timer.Start(); // Time the operation
296  }
297  mpi.DivideTasks(mpi.m_id, this->m_data.m_fockSpaceDim, mpi.m_nbrProcs, &mpi.m_firstTask, &mpi.m_lastTask, false);
298  fock_t lowerRowLimit = std::max((fock_t)mpi.m_firstTask, mpi.m_firstTask+minRow);
299  fock_t upperRowLimit = std::min((fock_t)mpi.m_lastTask, mpi.m_firstTask+maxRow);
300  // Reserve a memory block with which to retrieve the momentum conserving list of k-values
301  kState_t kRetrieveBuffer[termTables->GetMaxKCount()];
302  std::vector<fock_t>::const_iterator it_state;
303  fock_t currState = this->m_fockBasis.GetFirstState(lowerRowLimit, this->m_data.m_nbrParticles, it_state);
304  utilities::LoadBar progress;
305  if(0 == mpi.m_id) // FOR THE MASTER NODE
306  {
307  progress.Initialize(this->m_data.m_nodeDim);
308  }
309  // Calculate the sum over Vkkkk terms
310  for(int i=lowerRowLimit; i<=upperRowLimit; ++i)
311  {
312  fock_t inState = currState;
313  // Determine the k1,k2,k3,k4 orbitals that lead to a non-zero contribution,
314  // First, we can restrict the sum to k3,k4 values that do no annihilate
315  // the inState and that satisfy k3!=k4
316  fock_t tempK4 = inState;
317  const fock_t mask = ((fock_t)1 << this->m_data.m_nbrOrbitals)-1;
318  while(tempK4!=0) // N iterations for N particles
319  {
320  fock_t k4Occupied = tempK4 & -tempK4; // Isolate the right-most bit
321  kState_t k4 = utilities::binary::HammingWeight64(k4Occupied-1);
322  // Now process the in state with the k4 orbital removed
323  fock_t tempK3 = inState ^ k4Occupied;
324  while(tempK3!=0) // N iterations for N particles
325  {
326  fock_t k3Occupied = tempK3 & -tempK3; // Isolate the right-most bit
327  kState_t k3 = utilities::binary::HammingWeight64(k3Occupied-1);
328  if(k3 != k4) // For fermions, a squared creation operator gives zero
329  {
330  // Now we iterate over the possible values of k2
331  // that do not occur in the inState once k3 and k4 are removed
332  fock_t tempK2 = mask ^ (tempK3 ^ k3Occupied);
333  while(tempK2!=0)
334  {
335  fock_t k2Occupied = tempK2 & -tempK2; // Isolate the right-most bit
336  kState_t k2 = utilities::binary::HammingWeight64(k2Occupied-1);
337  // Get the corresponding list of k1 values
338  iSize_t nbrK1;
339  termTables->GetK1(kRetrieveBuffer, nbrK1, k2, k3, k4);
340  // Loop over the possible k1 values
341  for(iSize_t j=0; j<nbrK1; ++j)
342  {
343  kState_t k1 = kRetrieveBuffer[j];
344  fock_t k1Occupied = (fock_t)1 << k1;
345  // Restrict to cases that give non zero
346  // matrix elements, using the properties
347  // of spinless fermion operators
348  // If k1 != k3 or k4 then k1 must not occur in the
349  // inState and if k2 != k3 or k4 then k2 must
350  // not occur in the inState
351  if(
352  (!(k1Occupied & inState) || ((k1Occupied & k3Occupied) || (k1Occupied & k4Occupied))) &&
353  (!(k2Occupied & inState) || ((k2Occupied & k3Occupied) || (k2Occupied & k4Occupied))) &&
354  !(k1Occupied & k2Occupied)
355  )
356  {
357  // Evaluate the c^+ c^+ c c operator acting on the current state
358  int sign;
359  fock_t outState = utilities::Evaluate_CdCdCC(sign, k1Occupied, k2Occupied, k3Occupied,
360  k4Occupied, inState,
361  this->m_data.m_highestOrbital);
362  if(0 != sign && outState>=inState)
363  {
364  // If non zero then calculate the matrix element
365  // and find the index of the outState
366  fock_t outIndex = this->m_fockBasis.FindFockStateIndex(mpi.m_firstTask, outState);
367  if(outIndex < this->m_data.m_fockSpaceDim)
368  {
369  T matrixElement = (double)sign*termTables->GetVkkkk(k1, k2, k3, k4);
370  // Add the matrix element to the matrix
371  if((utilities::is_same<dcmplx,T>::value && abs(matrixElement)>_SPARSE_ZERO_TOL_) || (utilities::is_same<double, T>::value && fabs(matrixElement)>_SPARSE_ZERO_TOL_))
372  {
373  if(utilities::_DENSE_ == this->m_data.m_storageMethod)
374  {
375  this->m_fullMatrix[(i-mpi.m_firstTask)*this->m_data.m_fockSpaceDim+outIndex] += matrixElement;
376  }
377  else if(utilities::_SPARSE_MAPPED_ == this->m_data.m_storageMethod)
378  {
379  this->m_sparseMap.Insert((i-mpi.m_firstTask), outIndex) += matrixElement;
380  }
381  else if(utilities::_SPARSE_CRS_ == this->m_data.m_storageMethod)
382  {
383  this->m_crsSparseMatrix.Insert((i-mpi.m_firstTask),outIndex) += matrixElement;
384  }
385  }

387  }
388  }
389  }
390  }
391  tempK2 = tempK2 ^ k2Occupied; // Remove right-most bit and proceed to the next iteration
392  }
393  }
394  tempK3 = tempK3 ^ k3Occupied; // Remove right-most bit and proceed to the next iteration
395  }
396  tempK4 = tempK4 ^ k4Occupied; // Remove right-most bit and proceed to the next iteration
397  }
