70  //! \brief A function to perform matrix-vector multiplication on sparse
71  //! type matrices and vectors: y = alpha*A*x+beta*x where A is a Symmetric
72  //! or Hermitian sparse matrix
73  //!
74  //! This function calls an MKL BLAS routine to perform Hermitian
75  //! matrix-vector multiplication. If MKL is not available, use a naive
76  //! and slow back up method.
77  //////////////////////////////////////////////////////////////////////////////////
78  template <typename T>
79  void SymmetricMatrixVectorMultiply(
80  T* x, //!< Input vector
81  T* y, //!< Output vector
82  int dim, //!< Dimension of the vectors/square matrix
83  utilities::CrsSparseMatrix<T>* matrix,
84  //!< Address of the sparse CRS format matrix
85  char* UPLO, //!< Specify 'U' if upper triangular is stored,
86  //!< of 'L' if lower triangular is stored
87  T ALPHA, //!< Matrix scale factor
88  T BETA) //!< Additional vector scale factor
89  {
90  ////////////////////////////////////////////////////////////////////////////////
91  #if _ENABLE_INTEL_MKL_
92  ////// Declare variables to pass to mkl blas routine
93  // void mkl_?csrmv(char* TRANSA,int* M,int* K,T* ALPHA,char* MATDESCRA,
94  // T* VAL,int* INDX,int* PNTRB,int* PNTRE,T* X,
95  // T* BETA,T* Y,int* LEN_MATDESCRA)
96  char TRANSA; // if TRANSA = 'N' then y = alpha*A*x+beta*y
97  // if TRANSA = 'T' then y = alpha*A'*x+beta*y
98  int M; // Number of rows in the matrix A
99  int K; // Number of columns in the matrix A
100  char MATDESCRA[6]; // Matrix description (see Intel's website
101  // for possible values)
102  T* VAL; // Array containing non-zero elements
103  int* INDX; // Array containing column indices
104  int* PNTRB; // Array of dimension M such that PNTRB[i] - PNTRB[0]
105  // is the first index of row i
106  int* PNTRE; // Array of dimension M such that PNTRE[i] - PNTRB[0] - 1
107  // Is the last index of row i
108  T* X; // Input vector
109  T* Y; // Output vector
110  int LEN_MATDESCRA; // Additional argument to pass the length of the
111  // MATDESCRA character array
112  ////// Set parameter values
113  TRANSA = 'N'; // No transpose
114  M = dim;
115  K = dim;
116  MATDESCRA[0] = 'H'; // Hermitian matrix
117  MATDESCRA[1] = *UPLO; // Upper or lower triangular specified
118  MATDESCRA[2] = 'N'; // Not a unit matrix
119  MATDESCRA[3] = 'C'; // Use C-syle matrix indexing
120  VAL = matrix->GetData();
121  INDX = matrix->GetCols();
122  PNTRB = matrix->GetRowStarts();
123  PNTRE = matrix->GetRowStarts()+1;
124  X = x;
125  Y = y;
126  LEN_MATDESCRA = 6;
127  // Copy the x into the y vector. This allows us to calculate
128  // y = alpha Ax + beta*x (since the mkl function actually does
129  // y = alpha Ax + beta*y)
130  if(BETA!=0.0)
131  {
132  memcpy(Y, X, (long int)dim*sizeof(T));
133  }
134  if(utilities::is_same<T, dcmplx>::value)
135  {
136  mkl_zcsrmv(&TRANSA, &M, &K, &ALPHA, &MATDESCRA[0], VAL, INDX, PNTRB, PNTRE, X, &BETA, Y, &LEN_MATDESCRA);
137  }
138  else if(utilities::is_same<T, double>::value)
139  {
140  mkl_dcsrmv(&TRANSA, &M, &K, &ALPHA, &MATDESCRA[0], VAL, INDX, PNTRB, PNTRE, X, &BETA, Y, &LEN_MATDESCRA);
141  }
142  #else
143  // Naive (and slow) backup implementation
144  for(long int i=0; i<dim; ++i)
145  {
146  y[i] = BETA*x[i];
147  }
148  int* p_rowStarts = matrix->GetRowStarts();
149  int* p_cols = matrix->GetCols();
150  T* p_data = matrix->GetData();
151  for(int i=0; i<matrix->GetNbrRows(); ++i)
152  {
153  for(int j = p_rowStarts[i]; j<p_rowStarts[i+1]; ++j)
154  {
155  int tmpCol = p_cols[j];
156  T tmpElement = p_data[j];
157  y[i] += ALPHA*tmpElement*x[tmpCol];
158  // Take into account that our matrix is Hermitian
159  if(i !=tmpCol)
160  {
161  y[tmpCol] += ALPHA*std::conj(tmpElement)*x[i];
162  }
163  }
164  }
165  #endif
166  ////////////////////////////////////////////////////////////////////////////////
167  return;
168  }

171  //! \brief A function to perform a parallel matrix-vector multiplication on sparse
172  //! type matrices and vectors: y = A*x where A is a symmetric or
173  //! Hermitian sparse matrix
174  //!
175  //! This function is optimized to try to reduce inter-node communication.
176  //!
177  //! NOTE 1: The function expects that the input and output vectors will be
178  //! distributed over a given number of nodes. The dataDistribution
179  //! argument specifies the dimension of the part of the input/output
180  //! vector stored on the each node (and is identical on all nodes).
181  //!
182  //! NOTE 2: In order for this function to work properly, the sparse matrix
183  //! information must be updated with the PrepareForParallelMultiplication
184  //! function.
185  //!
186  //! The picture to have in mind is as follows. We solve Y = A X, where
187  //! the memory distribution of the vectors/matrix is as:
188  //!
189  //! +------------+
190  //! Y0 | A0 | X0 } dataDistribution[0]
191  //! | +--------+
192  //! Y1 | | A1 | X1 } dataDistribution[1]
193  //! | | +----+
194  //! Y2 | | | A2 | X2 } dataDistribution [2]
195  //!
196  //! Where the first dataDistribution[0] rows of the matrix are stored on
197  //! node 0, and so on. Note that the Hermitian structure of the matrix
198  //! means that the first n rows also give the first n columns.
199  //!
200  //! The matrix A0 can be thought of as a NxN matrix (N the total dimension)
201  //! with zeros where A1, A2 are labelled. A1 is a (N-dataDistribution[0])x
202  //! (N-dataDistribution[0]) matrix with zeros where A2 is labelled, and so on.
203  //! Appropriate restructuring is done by PrepareForParallelMultiplication.
204  //!
205  //! The matrix-vector multiplication can be done like this:
206  //!
207  //! Y = A0.[X0,X1,X2] + A1.[X1,X2] + A2.[X2]
208  //!
209  //! Where e.g. [X0,X1,X2] is a vector combining X0,X1,X2 and where the
210  //! output vectors of each matrix multiplication are appropriately padded
211  //! with zeros when summed.
212  //!
213  //! Assuming that the As take up a lot of memory, we focus on passing
214  //! the minimal amount of X around. Note also that the result in e.g Y0
215  //! does not depend on A1.[X1,X2] or A2.[X2] etc, so we can also
216  //! minimise the amount of Y passed around.
217  //////////////////////////////////////////////////////////////////////////////////
218  template <typename T>
219  void ParallelSymmetricMatrixVectorMultiply(
220  T* x, //!< Input vector on the current node
221  T* y, //!< Output vector on the current node
222  std::vector<int>* dataDistribution, //!< A vector containing a list
223  //! of the number of matrix rows
224  //! stored on each node
225  utilities::CrsSparseMatrix<T>* matrix,
226  //!< Address of the sparse CRS format matrix
227  char* UPLO, //!< Specify 'U' if upper triangular is stored,
228  //!< of 'L' if lower triangular is stored
229  T ALPHA, //!< Matrix scale factor
230  T BETA, //!< Additional vector scale factor
231  std::vector<MPI_Comm>* commGroups, //!< List of processor groups
232  //! required for minimal
233  //! vector communication
234  T* inputVectorBuffer, //!< Buffer for MPI distribution of X vector
235  T* outputVectorBuffer, //!< Buffer for MPI accumulation of Y vector
236  const int bufferDim, //!< Size of the input/output buffers
237  const MpiWrapper& mpi) //!< Instance of the MPI wrapper class
238  {
239  // The vector scale factor BETA is only used on the master node,
240  // and should be set to zero everywhere else
241  if(0 != mpi.m_id)
242  {
243  BETA = 0.0;
244  }
245  // To implement this, we define a set of M-1 processor groups,
246  // where M is the number of processors. The 1st group contains
247  // all M nodes, the second group contains M-1 nodes, etc.
248  // The first step is to put exactly the required amount of the input
249  // vector on each node. Node i requires only Xi, Xi+1 and so on,
250  // but not Xi-1 etc.
251  int cumulativeBcast = 0;
252  for(int i = 0; i<mpi.m_nbrProcs; ++i)
253  {
254  if(mpi.m_id == i)
255  {
256  memcpy(inputVectorBuffer, x, (*dataDistribution)[i]*sizeof(T));
257  }
258  if(0 != i)
259  {
260  // Broadcast the contents of the buffer to the preceding i nodes
261  // only, by splitting up the MPI communicator into 2 groups -
262  // one for the preceding nodes, one for the rest
263  int whichGroup = (i>=mpi.m_id ? 0 : 1);
264  // Only call the Bcast function on one of the processor
265  // groups - this saves a factor of 2 in communication time!
266  if(0 == whichGroup)
267  {
268  MPI_Bcast(inputVectorBuffer+cumulativeBcast, (*dataDistribution)[i], mpi.GetType<T>(), i, (*commGroups)[i]);
269  }
270  }
271  if(mpi.m_id <= i)
272  {
273  cumulativeBcast += (*dataDistribution)[i];
274  }
275  }
276  // Now inputVectorBuffer on node i stores [Xi,Xi+1,...]
277  // Next we can perform a serial matrix-vector multiplication
278  // i.e. the form Ai.[Xi,Xi+1,...]. The result will be stored in
279  // the outputVectorBuffer.
280  SymmetricMatrixVectorMultiply(inputVectorBuffer, outputVectorBuffer, bufferDim, matrix, UPLO, ALPHA, BETA);
281  // Finally we re-distribute the outputVectorBuffer to generate
282  // the correct Y vector on each node. This step performs the
283  // sum e.g. Y = A1.[X1,X2,X3] + A2.[X2,X3] + A3.[X3]
284  // Using an MPI reduce algorithm
285  int cumulativeReduce = 0;
286  for(int i = 0; i<mpi.m_nbrProcs; ++i)
287  {
288  if(0 != i)
289  {
290  int whichGroup = (i>=mpi.m_id ? 0 : 1);
291  // Only call the Reduce function on one of the processor
292  // groups - this saving a factor of 2 in communication time!
293  if(0 == whichGroup)
294  {
295  // Note that the inputVectorBuffer is recycled
296  // as working memory for the mpi.Reduce function
297  MPI_Status status;
298  mpi.Reduce<T>(outputVectorBuffer+cumulativeReduce,inputVectorBuffer,
299  (*dataDistribution)[i], i, (*commGroups)[i], status);
300  }
301  }
302  if(mpi.m_id == i)
303  {
304  memcpy(y, outputVectorBuffer, (*dataDistribution)[i]*sizeof(T));
305  }
306  if(mpi.m_id <= i)
307  {
308  cumulativeReduce += (*dataDistribution)[i];
309  }
310  }
311  return;
312  }
