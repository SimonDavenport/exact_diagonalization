179  void ToFileBase(
180  const std::string fileName, //!< File name
181  const io::fileFormat_t format, //!< Format of file
182  const iSize_t nbrLabels, //!< Number of quantum number labels

193  {
194  iSize_t strSize = m_fileHeader.size();
195  f_out.write((char*)&(strSize), sizeof(iSize_t));
196  f_out.write(m_fileHeader.c_str(), strSize);
197  f_out.write((char*)&nbrLabels, sizeof(iSize_t));
198  f_out.write((char*)&m_kMax, sizeof(iSize_t));
199  if(2 == nbrLabels)
200  {
201  f_out.write((char*)m_vTable.data(), m_vTable.size()*sizeof(T));
202  }
203  else if(4 == nbrLabels)
204  {
205  f_out.write((char*)m_kTable.data(), m_kTable.size()*sizeof(kState_t));
206  f_out.write((char*)m_vTable.data(), m_vTable.size()*sizeof(T));
207  }
208  }
209  else if(io::_TEXT_ == format)
210  {
211  f_out << m_fileHeader << "\n";
212  f_out << nbrLabels << "\n";
213  f_out << m_kMax << "\n";
214  if(2 == nbrLabels)
215  {
216  for(kState_t k1=0; k1<m_kMax; ++k1)
217  {
218  std::string streamData = utilities::ToStream(m_vTable[k1]);
219  f_out << k1 << "\t" << k1 << "\t" << streamData <<"\n";
220  }
221  }
222  else if(4 == nbrLabels)
223  {
224  kState_t k1;
225  for(kState_t k4=0; k4<m_kMax; ++k4)
226  {
227  for(kState_t k3=0; k3<m_kMax; ++k3)
228  {
229  for(kState_t k2=0; k2<m_kMax; ++k2)
230  {
231  iSize_t dummy;
232  this->GetK1(&k1, dummy, k2, k3, k4);
233  f_out << k1 << "\t" << k2 << "\t" << k3 << "\t" << k4;
234  std::string streamData = utilities::ToStream(this->GetVkkkk(k1, k2, k3, k4));
235  f_out << "\t" << streamData <<"\n";
236  }
237  }
238  }
239  }
240  }
241  f_out.close();
242  }
243  }
244  MPI_Barrier(mpi.m_comm);
245  mpi.ExitFlagTest();

248  //!
249  //! Import the term table from a file
250  //!
251  void FromFileBase(
252  const std::string fileName, //!< File name
253  const io::fileFormat_t format, //!< Format of file
254  utilities::MpiWrapper& mpi) //!< Instance of the MPI wrapper class
255  {
256  if(0 == mpi.m_id) // FOR THE MASTER NODE
257  {
258  std::ifstream f_in;
259  utilities::GenFileStream(f_in, fileName, format, mpi);
260  if(!mpi.m_exitFlag)
261  {
262  iSize_t nbrLabels = 0;
263  std::string fileHeader;
264  if(io::_BINARY_ == format)
265  {
266  iSize_t strSize;
267  f_in.read(reinterpret_cast<char*>(&strSize), sizeof(iSize_t));
268  fileHeader.resize(strSize);
269  f_in.read(&fileHeader[0], strSize);
270  if(m_fileHeader != fileHeader)
271  {
272  std::cerr << "\n\tFile read in error - wrong file header " << fileHeader << std::endl;
273  mpi.m_exitFlag = true;
274  f_in.close();
275  goto escape;
276  }
277  f_in.read(reinterpret_cast<char*>(&nbrLabels), sizeof(iSize_t));
278  f_in.read(reinterpret_cast<char*>(&m_kMax), sizeof(iSize_t));
279  this->Initialize(m_kMax, mpi);
280  if(2 == nbrLabels)
281  {
282  f_in.read(reinterpret_cast<char*>(m_vTable.data()), m_vTable.size()*sizeof(T));
283  }
284  else if(4 == nbrLabels)
285  {
286  f_in.read(reinterpret_cast<char*>(m_kTable.data()), m_kTable.size()*sizeof(kState_t));
287  f_in.read(reinterpret_cast<char*>(m_vTable.data()), m_vTable.size()*sizeof(T));
288  }
289  }
290  else if(io::_TEXT_ == format)
291  {
292  getline(f_in, fileHeader);
293  if(m_fileHeader != fileHeader)
294  {
295  std::cerr << "\n\tFile read in error - wrong file header " << fileHeader << std::endl;
296  mpi.m_exitFlag = true;
297  f_in.close();
298  goto escape;
299  }
300  f_in >> nbrLabels;
301  f_in >> m_kMax;
302  this->Initialize(m_kMax, mpi);
303  if(2 == nbrLabels)
304  {
305  kState_t k1;
306  kState_t k2;
307  for(iSize_t i=0; i<m_vTable.size(); ++i)
308  {
309  f_in >> k1 >> k2;
310  utilities::FromStream(f_in, m_vTable[i]);
311  }
312  }
313  else if(4 == nbrLabels)
314  {
315  for(kState_t k4=0; k4<m_kMax; ++k4)
316  {
317  for(kState_t k3=0; k3<m_kMax; ++k3)
318  {
319  for(kState_t k2=0; k2<m_kMax; ++k2)
320  {
321  kState_t k1, dummy;
322  f_in >> k1 >> dummy >> dummy >> dummy;
323  T Vkkkk;
324  utilities::FromStream(f_in, Vkkkk);
325  this->SetK1(k1, k2, k3, k4);
326  this->SetVkkkk(Vkkkk, k1, k2, k3, k4);
327  }
328  }
329  }
330  }
331  }
332  f_in.close();
333  }
334  }
335  escape:
336  {}
337  mpi.ExitFlagTest();
338  mpi.Sync(&m_kMax, 1, 0);
339  if(0 != mpi.m_id) // FOR ALL OTHER NODES
340  {
341  this->Initialize(m_kMax, mpi);
342  }
343  MPI_Barrier(mpi.m_comm);
344  this->MpiSynchronize(0, mpi);
345  }
