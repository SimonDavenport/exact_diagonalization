217  //! \brief A template function to set the quartic hash table values from
218  //! any other class container storing the quartic terms.
219  //!
220  //! The function expects a class containing the methods:
221  //!
222  //! - GetK1(&k1,nbrK1,k2,k3,k4)
223  //!
224  //! Returns k1 for a given k2,k3,k4 momentum
225  //!
226  //! - GetVkkkk(k1,k2,k3,k4)
227  //!
228  //! Returns a single Vkkkk coefficient for a given k1,k2,k3,k4 set of momenta
229  ////////////////////////////////////////////////////////////////////////////////
230  template <class C>
231  void SetQuarticFromArray(
232  C* quarticArray) //!< Class container for quartic term array
233  {
234  for(kState_t k2=0; k2<m_kMax; ++k2)
235  {
236  for(kState_t k3=0; k3<m_kMax; ++k3)
237  {
238  for(kState_t k4=0; k4<m_kMax; ++k4)
239  {
240  kState_t k1;
241  iSize_t nbrK1;
242  quarticArray->GetK1(&k1, nbrK1, k2, k3, k4);
243  m_kTable.Insert(k1, utilities::Key(k2, k3, k4));
244  m_vTable.Insert(utilities::Key(k1, k2, k3, k4)) = quarticArray->GetVkkkk(k1, k2, k3, k4);
245  }
246  }
247  }
