235  ////////////////////////////////////////////////////////////////////////////////
236  //! \brief Populate the difference equation relating Bloch coefficients
237  //! with different reciprocal lattice indices
238  //!
239  //! The matrix is stored with spin up /spin down components alternating
240  ////////////////////////////////////////////////////////////////////////////////
241  void NonInteractingOflModel::PopulateDifferenceMatrix()
242  {
243  if(m_differenceMatrixPopulated)
244  {
245  delete[] m_matrix;
246  }
247  // The difference equation includes contributions form both positive and
248  // negative reciprocal lattice points.
249  // The dimension of the difference equation matrix is given by the number of
250  // points in 2D k-space multiplied by the number of spin species
251  m_dim = this->GetDifferenceMatrixDimension();
252  m_matrix = new (std::nothrow) dcmplx[m_dim*m_dim];
253  const dcmplx alpha = m_params.m_V0*(3*cos(m_params.m_theta)*cos(m_params.m_theta)-1.0)/2.0;
254  const dcmplx beta = dcmplx(0,-sqrt(3.0)*m_params.m_V0*sin(m_params.m_theta)*sin(m_params.m_theta)/2.0);
255  const dcmplx gamma = m_params.m_epsilon*m_params.m_V0*cos(m_params.m_theta);
256  const dcmplx alphaPlusBeta = alpha + beta;
257  const dcmplx alphaMinusBeta = alpha - beta;
258  // Populate the matrix (upper triangular elements only)
259  for(iSize_t i=0; i<m_dim; ++i)
260  {
261  int lRow,mRow;
262  this->MapArrayToLattice(&lRow, &mRow, i/2);
263  for(iSize_t j=i; j<m_dim; ++j)
264  {
265  // Keep track of the book keeping in assigning zero /non-zero
266  // matrix elements
267  int lCol,mCol;
268  this->MapArrayToLattice(&lCol, &mCol, j/2);
269  const int deltaL = lCol - lRow;
270  const int deltaM = mCol - mRow;
271  const spin_t spinStateI = i&1 ? _SPIN_DOWN_ : _SPIN_UP_;
272  const spin_t spinStateJ = j&1 ? _SPIN_DOWN_ : _SPIN_UP_;
273  dcmplx element = 0.0;
274  if(i==j)
275  {
276  element = this->EvaluateDiagonalTerm(lCol, mCol, spinStateI);
277  }
278  else if(std::abs(deltaL) > 1 || std::abs(deltaM) > 1)
279  {
280  element = 0.0;
281  }
282  else
283  {
284  if(spinStateI==_SPIN_UP_ && spinStateJ==_SPIN_UP_)
285  {
286  // spin up - spin up terms
287  if(deltaL == 0 && deltaM == 1)
288  {
289  element = alphaPlusBeta;
290  }
291  else if(deltaL == 0 && deltaM == -1)
292  {
293  element = alphaMinusBeta;
294  }
295  else if(deltaL == 1 && deltaM == 0)
296  {
297  element = alphaMinusBeta;
298  }
299  else if(deltaL == -1 && deltaM == 0)
300  {
301  element = alphaPlusBeta;
302  }
303  else if(deltaL == 1 && deltaM == -1)
304  {
305  element = alphaPlusBeta;
306  }
307  else if(deltaL == -1 && deltaM == 1)
308  {
309  element = alphaMinusBeta;
310  }
311  }
312  else if(spinStateI==_SPIN_UP_ && spinStateJ==_SPIN_DOWN_)
313  {
314  // spin up - spin down terms
315  if(deltaL == 0 && deltaM == -1)
316  {
317  element = gamma;
318  }
319  else if(deltaL == -1 && deltaM == 0)
320  {
321  element = gamma;
322  }
323  else if(deltaL == 0 && deltaM == 0)
324  {
325  element = gamma;
326  }
327  }
328  else if(spinStateI==_SPIN_DOWN_ && spinStateJ==_SPIN_UP_)
329  {
330  // spin down - spin up terms
331  if(deltaL == 0 && deltaM == 1)
332  {
333  element = gamma;
334  }
335  else if(deltaL == 1 && deltaM == 0)
336  {
337  element = gamma;
338  }
339  else if(deltaL == 0 && deltaM == 0)
340  {
341  element = gamma;
342  }
343  }
344  else
345  {
346  // spin down spin down terms
347  if(deltaL == 0 && deltaM == 1)
348  {
349  element = alphaMinusBeta;
350  }
351  else if(deltaL == 0 && deltaM == -1)
352  {
353  element = alphaPlusBeta;
354  }
355  else if(deltaL == 1 && deltaM == 0)
356  {
357  element = alphaPlusBeta;
358  }
359  else if(deltaL == -1 && deltaM == 0)
360  {
361  element = alphaMinusBeta;
362  }
363  else if(deltaL == 1 && deltaM == -1)
364  {
365  element = alphaMinusBeta;
366  }
367  else if(deltaL == -1 && deltaM == 1)
368  {
369  element = alphaPlusBeta;
370  }
371  }
372  }
373  m_matrix[i*m_dim+j] = element;
374  }
375  }
376  m_differenceMatrixPopulated = true;
377  }

405  //!
406  //! Evaluate diagonal term in difference matrix
407  //!
408  double NonInteractingOflModel::EvaluateDiagonalTerm(
409  const int l, //!< number of G1 lattice vectors
410  const int m, //!< number of G2 lattice vectors
411  const spin_t spinState) //!< Spin state of the matrix element
412  const
413  {
414  // Diagonal terms only contain kinetic energy
415  // (shifted by - 0.5 Sigma_z due to the gauge transformation)
416  double temp1 = m_G.GetKx() + l*reciprocalLattice::G1.GetKx() + m*reciprocalLattice::G2.GetKx();
417  double temp2 = m_G.GetKy() + l*reciprocalLattice::G1.GetKy() + m*reciprocalLattice::G2.GetKy();
418  if(spinState==_SPIN_UP_)
419  {
420  temp2 += 1.0/2.0;
421  }
422  else
423  {
424  temp2 -= 1.0/2.0;
425  }
426  return this->GetRecoilEnergy()*(temp1*temp1 + temp2*temp2 + 6.0*m_params.m_V0);
427  }

429  //!
430  //! Diagonalize the difference equation to determine the energy bands
431  //! and the values of the Bloch coefficients
432  //!
433  void NonInteractingOflModel::DiagonalizeDifferenceMatrix()
434  {
435  if(m_differenceMatrixPopulated)
436  {
437  // Reset previous memory allocation
438  if(m_blochCoefficients!=0) delete[] m_blochCoefficients;
439  if(m_eigenvalues!=0) delete[] m_eigenvalues;
440  m_blochCoefficients = new (std::nothrow) dcmplx[m_dim*m_nbrBands];
441  m_eigenvalues = new (std::nothrow) double[m_nbrBands];
442  if(m_params.m_V0==0.0)
443  {}
444  else
445  {
446  // Generate a copy of the matrix to be diagonalized
447  // (since the diagonalization routine will destroy the
448  // original matrix, which we want to keep)
449  dcmplx *matrix = new dcmplx[m_dim*m_dim];
450  memcpy(matrix, m_matrix, m_dim*m_dim*sizeof(dcmplx));
451  utilities::linearAlgebra::DiagonalizeSymmetricMatrix<dcmplx>(matrix, m_blochCoefficients, m_eigenvalues,
452  m_dim, m_nbrBands, 'U');
453  delete[] matrix;
454  }
455  }
456  else
457  {}
458  m_bandsCalculated = true;
459  return;
460  }

541  ////////////////////////////////////////////////////////////////////////////////
542  //! \brief Return the nth eigenvector of the difference Hamiltonian, which
543  //! describes the coefficients appearing in the Bloch wave function labelling
544  //! each reciprocal lattice point.
545  //!
546  //! Makes a copy by value of the full eigenvector for the nth band and spin s.
547  //! Note that the eigenvector is stored as a staggered array, altenating between
548  //! spin-up and spin down components
549  ////////////////////////////////////////////////////////////////////////////////
550  void NonInteractingOflModel::GetBlochCoefficients(
551  dcmplx* eigenvector, //!< Block of memory to copy the eigenvector into
552  const iSize_t n, //!< Band index
553  const spin_t s, //!< Spin index (0 or 1) of enum type spin

555  const

562  dcmplx* p_bloch = m_blochCoefficients;
563  p_bloch += n*m_dim + s;
564  for(iSize_t i=0; i<m_dim/2; ++i, ++p_vector, p_bloch+=2)
565  {
566  if(_CONJUGATE_ == conjugate)
567  {
568  *p_vector = std::conj(*p_bloch);
569  }
570  else
571  {
572  *p_vector = *p_bloch;
573  }
574  }
575  }
576  }
577  else
578  {
579  return;
580  }
581  return;
582  }

597  //!
598  //! Calculate the value of the sigma^z operator applied to the Block
599  //! state. This operator will have the effect of muliplying all
600  //! spin-down amplitudes by a minus sign. The spin magnitude is taken to be 1
601  //!
602  double NonInteractingOflModel::ApplySigmaZ(
603  const iSize_t n) //!< Band index
604  const
605  {
606  dcmplx* p_bloch = m_blochCoefficients + n*m_dim;
607  dcmplx sigmaZ = 0.0;
608  for(iSize_t i=0; i<m_dim; ++i, ++p_bloch)
609  {
610  if(i&1) // odd terms are associated with spin down
611  {
612  sigmaZ -= std::conj(*(p_bloch)) * *(p_bloch);
613  }
614  else
615  {
616  sigmaZ += std::conj(*(p_bloch)) * *(p_bloch);
617  }
618  }
619  return real(sigmaZ);
620  }
