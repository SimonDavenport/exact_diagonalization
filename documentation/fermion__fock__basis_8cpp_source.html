226  ////////////////////////////////////////////////////////////////////////////////
227  //! \brief Allocate memory to store the full Fock space. Generate the Fock space
228  //! and store it in the m_basis data structure.
229  //!
230  //! Optionally, a constraint function can be provided which will be applied
231  //! to place a restriction on the allowed Fock states (for instance a quantum
232  //! number constraint)
233  //!
234  //! This function MUST be called on all nodes if run in parallel. When run
235  //! in parallel, only the basis of states after the starting node index is
236  //! stored - this is because for binary search of a Hermitian matrix element,
237  //! only the upper triangular part of the matrix is required for the search.
238  //!
239  //! \return true if the Fock space dimension is 0, false otherwise
240  ////////////////////////////////////////////////////////////////////////////////
241  bool FermionFockBasis::GenerateFockSpace(
242  const iSize_t nbrParticles, //!< Number of particles
243  const iSize_t nbrOrbitals, //!< Number of orbitals
244  const std::function<bool(const fock_t state)>& Constraint,
245  //!< A constraint function to test whether the
246  //!< state "state" corresponds to a given quantum
247  //!< number or not
248  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
249  {
250  if(0 == mpi.m_id) // FOR THE MASTER NODE
251  {
252  utilities::cout.AdditionalInfo()<<"\n\t- BUILDING FOCK BASIS"<<std::endl;
253  }
254  // Determine the set of states corresponding to the quantum number
255  // sector imposed by the constraint function.
256  // We can call the constraint function in parallel to save time,
257  // however there will not in general be an equal distribution of
258  // allowed states over all nodes
259  fock_t fullFockSpaceDim = this->GetFullFockSpaceDimension(nbrParticles,nbrOrbitals);
260  mpi.DivideTasks(mpi.m_id, fullFockSpaceDim, mpi.m_nbrProcs, &mpi.m_firstTask, &mpi.m_lastTask, false);
261  fock_t fullNodeDim = mpi.m_lastTask - mpi.m_firstTask + 1;
262  fock_t sectorCounter = 0;
263  fock_t sectorTotalDimension = 0;
264  {
265  // Generate first state on each node
266  fock_t currState = utilities::binary::GenerateHammingNumber(nbrParticles,mpi.m_firstTask);
267  // Count the set of states in quantum number sector specified by the Constraint function
268  for(fock_t i=0; i<fullNodeDim; ++i)
269  {
270  if(Constraint(currState))
271  {
272  ++sectorCounter;
273  }
274  currState = utilities::binary::NextHammingNumber64(currState);
275  }
276  // Update the dimension variable to store the total number of allowed
277  // states and count the cumulative number of states on each node
278  MPI_Reduce(&sectorCounter, &sectorTotalDimension, 1, mpi.GetType<fock_t>(), MPI_SUM, 0, mpi.m_comm);
279  mpi.Sync(&sectorTotalDimension, 1, 0);
280  if(0 == mpi.m_id) // FOR THE MASTER NODE
281  {
282  utilities::cout.SecondaryOutput()<<"\n\t\tFOCK BASIS CONTAINS "<<sectorTotalDimension<<" STATE(S)"<<std::endl;
283  }
284  // Don't proceed if the Fock space dimension is 0
285  if(0 == sectorTotalDimension)
286  {
287  return true;
288  }
289  }
290  // Next store a list of the allowed Fock states using the same parallelization
291  fock_t* fockStateBuffer = 0;
292  {
293  fock_t* allowedFockStates = 0;
294  // Allocate memory to store part of the Fock basis on each node, and
295  // on the master node allocate enough space to store the full basis
296  allowedFockStates = new (std::nothrow) fock_t[sectorCounter];
297  fock_t* p_list = allowedFockStates;
298  fock_t currState = utilities::binary::GenerateHammingNumber(nbrParticles, mpi.m_firstTask);
299  for(fock_t i=0; i<fullNodeDim; ++i)
300  {
301  if(Constraint(currState))
302  {
303  *p_list = currState;
304  ++p_list;
305  }
306  currState = utilities::binary::NextHammingNumber64(currState);
307  }
308  // MPI gather the full list on the master node
309  if(0 == mpi.m_id) // FOR THE MASTER NODE
310  {
311  fockStateBuffer = new fock_t[sectorTotalDimension];
312  }
313  MPI_Status status;
314  mpi.Gather<fock_t>(allowedFockStates, sectorCounter, fockStateBuffer, sectorTotalDimension,
315  0, mpi.m_comm, status);
316  delete[] allowedFockStates;
317  }
318  MPI_Barrier(mpi.m_comm);
319  // If running in parallel, then to enable the binary search of states
320  // obtained after operating with c^+c... operators, we need only store those
321  // states that are greater than the diagonal (this is due to Hermiticity).
322  // Let's now calculate a cumulative total of the full number of rows that
323  // will be stored on each node
324  {
325  mpi.DivideTasks(mpi.m_id, sectorTotalDimension, mpi.m_nbrProcs, &mpi.m_firstTask, &mpi.m_lastTask, true);
326  fock_t cumulativeDim = sectorTotalDimension - mpi.m_firstTask;
327  m_firstState = mpi.m_firstTask;
328  // Allocate memory to store the allowed states
329  utilities::cout.DebuggingInfo()<<"\n\t\tNODE "<<mpi.m_id<<" ALLOCATING "<<(cumulativeDim*(sizeof(fock_t))/(1024.0*1024.0));
330  utilities::cout.DebuggingInfo()<<" MB TO STORE FOCK SPACE"<<std::endl;
331  m_basis.resize(cumulativeDim);
332  // Redistribute the fockStateBuffer over the other nodes
333  fock_t taskList[mpi.m_nbrProcs];
334  fock_t nodeDimList[mpi.m_nbrProcs];
335  MPI_Gather(&m_firstState, 1, mpi.GetType<fock_t>(), taskList, 1, mpi.GetType<fock_t>(), 0, mpi.m_comm);
336  MPI_Gather(&cumulativeDim, 1, mpi.GetType<fock_t>(), nodeDimList, 1, mpi.GetType<fock_t>(), 0, mpi.m_comm);
337  if(0 == mpi.m_id) // FOR THE MASTER NODE
338  {
339  for(int i=1; i<mpi.m_nbrProcs; ++i)
340  {
341  MPI_Send(fockStateBuffer+taskList[i], nodeDimList[i], mpi.GetType<fock_t>(), i, 20+i, mpi.m_comm);
342  }
343  memcpy(m_basis.data(), fockStateBuffer, nodeDimList[0]*sizeof(fock_t));
344  delete[] fockStateBuffer;
345  }
346  else
347  {
348  MPI_Recv(&m_basis[0], cumulativeDim, mpi.GetType<fock_t>(), 0, 20+mpi.m_id, mpi.m_comm, MPI_STATUS_IGNORE);
349  }
350  }
351  MPI_Barrier(mpi.m_comm);
352  m_basisStored = true;
353  return false;
354  }
