245 std::vector<diagonalization::iSize_t> GenerateSectorList(
246  boost::program_options::variables_map* optionList,
247  //!< Import command line option list to set sectors
248  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
249 {
250  std::vector<diagonalization::iSize_t> sectorList;
251  if(0 == mpi.m_id) // FOR THE MASTER NODE
252  {
253  bool blockDiagonalize;
254  GetOption(optionList, blockDiagonalize, "block-diagonalize", _AT_, mpi);
255  if(optionList->count("lz-sectors"))
256  {
257  GetOption(optionList, sectorList, "lz_sectors", _AT_, mpi);
258  }
259  else if(blockDiagonalize)
260  {
261  diagonalization::iSize_t startLz = 0;
262  diagonalization::iSize_t nbrParticles;
263  diagonalization::iSize_t nbrOrbitals;
264  diagonalization::iSize_t nbrLevels;
265  GetOption(optionList, nbrParticles, "nbr-particles", _AT_, mpi);
266  GetOption(optionList, nbrOrbitals, "nbr-orbitals", _AT_, mpi);
267  GetOption(optionList, nbrLevels, "nbr-levels", _AT_, mpi);
268  bool nbrParticlesOdd = nbrParticles & 1;
269  bool nbrOrbitalsOdd = nbrOrbitals & 1;
270  if(!nbrOrbitalsOdd && nbrParticlesOdd)
271  {
272  startLz = 1;
273  }
274  diagonalization::iSize_t max=0;
275  if(1==nbrLevels)
276  {
277  max = nbrParticles*(nbrOrbitals-nbrParticles);
278  }
279  else if(2==nbrLevels)
280  {
281  max = (nbrOrbitals-2)/2+1;
282  for(unsigned int i=2; i<=nbrParticles; ++i)
283  {
284  max += ((nbrOrbitals-2)/2+1-2*(i/2));
285  }
286  }
287  for(diagonalization::iSize_t lz = startLz; lz<=max; lz+=1)
288  {
289  sectorList.push_back(lz);
290  }
291  }
292  }
293  mpi.ExitFlagTest();
294  mpi.Sync(&sectorList, 0);
295  return sectorList;
296 }
