92  //!
93  //! Update the stored dimension of the quadratic term look-up table
94  //!
95  iSize_t QuarticTermTables::CalculateDim(
96  const kState_t kMax) //!< Number of k states
97  const
98  {
99  // In the quartic term table we store V_{k1,k2,k3,k4}, but with one of
100  // the k indices completely fixed by quantum number conservation
101  return (kMax+1)*std::pow(kMax,2)/2;
102  }

104  //!
105  //! Get the k1 value corresponding to a given k2,k3,k4 (where each k
106  //! represents a k1,ky pair).
107  //!
108  void QuarticTermTables::GetK1(
109  kState_t* kRetrieveBuffer, //!< Buffer to store retrieved k values
110  iSize_t& nbrK1, //!< Set the number of returned values
111  const kState_t k2, //!< k2 index
112  const kState_t k3, //!< k3 index
113  const kState_t k4) //!< k4 index
114  const
115  {
116  kState_t temp1 = k4;
117  kState_t temp2 = k3;
118  if(k4>k3)
119  {
120  // Swap and refer to the k3, k4 term in the table
121  temp1 = k3;
122  temp2 = k4;
123  }
124  nbrK1 = 1;
125  kRetrieveBuffer[0] = m_kTable[(temp1*(m_kMax) + temp2 - (temp1*(temp1+1))/2 )*m_kMax + k2];
126  }

149  //! \brief Return a quartic term coefficient specified by a given k1,k2,k3,k4
150  //!
151  //! NOTE: This function is ONLY valid when the k1,k2,k3,k4 indices form
152  //! a momentum conserving set - otherwise the coefficient does not exist
153  //! e.g. V_0001 does not correspond to a momentum conserving set of k1,k2,k3,k4
154  //! , but the GetVkkkk function will return a non-zero result nevertheless
155  //! , and that result will be incorrect!
156  ////////////////////////////////////////////////////////////////////////////////
157  dcmplx QuarticTermTables::GetVkkkk(
158  const kState_t k1, //!< k1 index
159  const kState_t k2, //!< k2 index
160  const kState_t k3, //!< k3 index
161  const kState_t k4) //!< k4 index
162  const
163  {
164  kState_t temp1 = k4;
165  kState_t temp2 = k3;
166  kState_t temp3 = k2;
167  if(k4>k3)
168  {
169  // Swap and refer to the k3, k4 term in the table (also requires swapping k1 and k2)
170  temp1 = k3;
171  temp2 = k4;
172  temp3 = k1;
173  }
174  return m_vTable[(temp1*(m_kMax) + temp2 - (temp1*(temp1+1))/2 )*m_kMax + temp3];
175  }
