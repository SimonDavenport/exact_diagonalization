107  //! \brief matrix-vector function involving recalculation of the matrix at
108  //! each step
109  //!
110  //! TODO debug this function - it does not currently work!

112  void Recalculate(
113  L1* quadraticLookUpTables, //!< Class container for look-up tables
114  L2* quarticLookUpTables, //!< Class container for look-up tables
115  T* x, //!< Input vector on the current node
116  T* y, //!< Output vector on the current node
117  utilities::MpiWrapper& mpi) //!< Instance of the mpi wrapper class
118  {
119  fock_t nbrRowsPerSweep = 2;
120  fock_t nbrSweeps = std::ceil((double)this->hamiltonian->m_data.m_nodeDim/nbrRowsPerSweep);
121  fock_t currRow = 0;
122  fock_t nbrNonZeros = 10000;
123  T* tempVector = new T[this->hamiltonian->m_data.m_nodeDim];
124  // Generate and store in CRS format a subset of all the matrix rows then
125  // perform a reduced matrix-vector multiplication for that subset of
126  // rows and store the outcome in a buffer
127  this->hamiltonian->m_crsSparseMatrix.Clear();
128  this->hamiltonian->m_crsSparseMatrix.Initialize(this->hamiltonian->m_data.m_nodeDim,
129  this->hamiltonian->m_data.m_fockSpaceDim, nbrNonZeros);
130  for(fock_t sweep=0; sweep<nbrSweeps; ++sweep, currRow+=nbrRowsPerSweep)
131  {
132  this->hamiltonian->Add_CdC_Terms(quadraticLookUpTables, currRow, currRow+nbrRowsPerSweep, mpi);
133  this->hamiltonian->Add_CdCdCC_Terms(quarticLookUpTables, currRow, currRow+nbrRowsPerSweep, mpi);
134  this->Simple(x, y, mpi);
135  for(fock_t i=0; i<this->hamiltonian->m_data.m_nodeDim; ++i)
136  {
137  tempVector[i] += y[i];
138  }
139  }
140  // Copy the accumulated vector into the output buffer
141  memcpy(y, tempVector.data(), tempVector.size()*sizeof(T));
142  delete[] tempVector;
143  }
